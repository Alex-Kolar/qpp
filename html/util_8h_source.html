185 // Kronecker product of 2 matrices, preserve return type
186 template<typename Scalar>
187 types::DynMat<Scalar> kron(const types::DynMat<Scalar> &A,
188  const types::DynMat<Scalar> &B)
189 {
190  // check zero-size
191  if (!internal::_check_nonzero_size(A))
192  throw Exception("kron", Exception::Type::MATRIX_ZERO_SIZE);
193 
194  // check zero-size
195  if (!internal::_check_nonzero_size(B))
196  throw Exception("kron", Exception::Type::MATRIX_ZERO_SIZE);
197 
198  size_t Acols = static_cast<size_t>(A.cols());
199  size_t Arows = static_cast<size_t>(A.rows());
200  size_t Bcols = static_cast<size_t>(B.cols());
201  size_t Brows = static_cast<size_t>(B.rows());
202 
203  types::DynMat<Scalar> result;
204  result.resize(Arows * Brows, Acols * Bcols);
205 
206  for (size_t j = 0; j < Acols; j++)
207 #pragma omp parallel for
208  for (size_t i = 0; i < Arows; i++)
209  result.block(i * Brows, j * Bcols, Brows, Bcols) = A.eval()(i, j)
210  * B.eval();
211 
212  return result;
213 
214 }

292  throw Exception("syspermute", Exception::Type::PERM_MISMATCH_DIMS);
293

304  cdims[i] = dims[i];

309  size_t jperm = 0;
310  for (size_t j = 0; j < dim; j++)
311  {
312  // compute the col multi-index

314 #pragma omp parallel for

317  iperm, jperm, A, result);

369 template<typename Scalar>
370 types::DynMat<Scalar> ptrace(const types::DynMat<Scalar> &A,
371  const std::vector<size_t> &subsys, const std::vector<size_t> &dims)
372 
373 {
374 // error checks
375 
376  // check zero-size
377  if (!internal::_check_nonzero_size(A))
378  throw Exception("ptrace", Exception::Type::MATRIX_ZERO_SIZE);
379 
380  // check square matrix
381  if (!internal::_check_square_mat(A))
382  throw Exception("ptrace", Exception::Type::MATRIX_NOT_SQUARE);
383 
384 // check that dims is a valid dimension vector
385  if (!internal::_check_dims(dims))
386  throw Exception("ptrace", Exception::Type::DIMS_HAVE_ZERO);
387 
388 // check that dims match the dimension of A
389  if (!internal::_check_dims_match_mat(dims, A))
390  throw Exception("ptrace", Exception::Type::DIMS_MISMATCH_MATRIX);
391 
392 // check that subsys are valid
393  if (!internal::_check_subsys(subsys, dims))
394  throw Exception("ptrace", Exception::Type::SUBSYS_MISMATCH_DIMS);
395 
396  size_t dim = static_cast<size_t>(A.rows());
397  size_t numsubsys = subsys.size(); // number of subsystems we trace out
398  size_t numdims = dims.size(); // total number of subsystems;
399  std::vector<size_t> perm(numdims, 0); // the permutation vector
400  std::vector<size_t> permdims; // the permuted dimensions
401 
402  types::DynMat<Scalar> result;
403 
404 // the total dimension of the traced-out subsystems
405  size_t dimsubsys = 1;
406  for (size_t i = 0; i < numsubsys; i++)
407  dimsubsys *= dims[subsys[i]];
408 
409  std::vector<size_t> sizeAB;
410  sizeAB.push_back(dim / dimsubsys);
411  sizeAB.push_back(dimsubsys);
412 
413 // construct the permutation that bring the traced-out subsystems to the end
414  size_t cnt0 = 0;
415  size_t cnt1 = 0;
416  for (size_t i = 0; i < numdims; i++)
417  {
418  // we find that i belongs to the subsystem
419  if (std::find(std::begin(subsys), std::end(subsys), i)
420  != std::end(subsys))
421  {
422  perm[numdims - numsubsys + cnt0] = i;
423  cnt0++;
424  }
425  else
426  {
427  perm[cnt1] = i;
428  cnt1++;
429  }
430  }
431 
432  return ptrace2(syspermute(A, perm, dims), sizeAB);
433 }
