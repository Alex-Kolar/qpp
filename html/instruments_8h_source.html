38  std::pair<std::vector<double>, std::vector<cmat>> measure(
39  const Eigen::MatrixBase<Derived> &A, const std::vector<cmat> &Ks)
40  {
41  const DynMat<typename Derived::Scalar> &rA = A;
42 
43  // EXCEPTION CHECKS
44  // check zero-size
45  if (!internal::_check_nonzero_size(rA))
46  throw Exception("qpp::measure()", Exception::Type::ZERO_SIZE);
47 
48  // check the Kraus operators
49  if (!internal::_check_nonzero_size(Ks))
50  throw Exception("qpp::measure()", Exception::Type::ZERO_SIZE);
51  if (!internal::_check_square_mat(Ks[0]))
52  throw Exception("qpp::measure()", Exception::Type::MATRIX_NOT_SQUARE);
53  if (Ks[0].rows() != rA.rows())
54  throw Exception("qpp::measure()", Exception::Type::DIMS_MISMATCH_MATRIX);
55  for (auto &&it : Ks)
56  if (it.rows() != Ks[0].rows() || it.cols() != Ks[0].rows())
57  throw Exception("qpp::measure()", Exception::Type::DIMS_NOT_EQUAL);
58  // END EXCEPTION CHECKS
59 
60  // probabilities
61  std::vector<double> prob(Ks.size());
62  // resulting states
63  std::vector<cmat> outstates(Ks.size());
64 
65  if (internal::_check_square_mat(rA)) // square matrix
66  {
67  for (std::size_t i = 0; i < Ks.size(); ++i)
68  {
69  cmat tmp;
70  tmp = Ks[i] * rA * adjoint(Ks[i]); // un-normalized
71  prob[i] = std::abs(trace(tmp)); // probability
72  if (prob[i] > eps)
73  outstates[i] = tmp / prob[i]; // normalized
74  }
75  }
76  else if (internal::_check_col_vector(rA)) // column vector
77  {
78  for (std::size_t i = 0; i < Ks.size(); ++i)
79  {
80  outstates[i] = ket::Zero(rA.rows());
81  ket tmp;
82  tmp = Ks[i] * rA; // un-normalized
83  // probability
84  prob[i] = std::abs((adjoint(tmp) * tmp).value());
85  if (prob[i] > eps)
86  outstates[i] = tmp / std::sqrt(prob[i]); // normalized
87  }
88  }
89  else
90  throw Exception("qpp::measure()",
91  Exception::Type::MATRIX_NOT_SQUARE_OR_CVECTOR);
92 
93  return std::make_pair(prob, outstates);
94 
95  }
