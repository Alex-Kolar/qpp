104 // Random unitary matrix
105 // ~3 times slower than Toby Cubitt's MATLAB's,
106 // because Eigen's QR algorithm is not parallelized
107 cmat randU(std::size_t D)
108 {
109  if (D == 0)
110  throw Exception("randU", Exception::Type::DIMS_INVALID);
111 
112  cmat X(D, D);
113 
114  X = 1 / std::sqrt(2.) * randn<cmat>(D, D);
115  Eigen::HouseholderQR<cmat> qr(X);
116 
117  cmat Q = qr.householderQ();
118  // phase correction so that the resultant matrix is
119  // uniformly distributed according to the Haar measure
120 
121  Eigen::VectorXcd phases = (rand<dmat>(D, 1)).cast<cplx>();
122  for (std::size_t i = 0; i < static_cast<std::size_t>(phases.rows()); i++)
123  phases(i) = std::exp((cplx) (2 * pi * 1_i * phases(i)));
124 
125  Q = Q * phases.asDiagonal();
126 
127  return Q;
128 }
