239  cmat randU(std::size_t D)
240  // ~3 times slower than Toby Cubitt's MATLAB corresponding routine,
241  // because 's QR algorithm is not parallelized
242  {
243  if (D == 0)
244  throw Exception("qpp::randU()", Exception::Type::DIMS_INVALID);
245 
246  cmat X(D, D);
247 
248  X = 1 / std::sqrt(2.) * randn < cmat > (D, D);
249  Eigen::HouseholderQR<cmat> qr(X);
250 
251  cmat Q = qr.householderQ();
252  // phase correction so that the resultant matrix is
253  // uniformly distributed according to the Haar measure
254 
255  Eigen::VectorXcd phases = (rand < dmat > (D, 1)).cast<cplx>();
256  for (std::size_t i = 0; i < static_cast<std::size_t>(phases.rows()); ++i)
257  phases(i) = std::exp((cplx) (2 * pi * 1_i * phases(i)));
258 
259  Q = Q * phases.asDiagonal();
260 
261  return Q;
262  }
