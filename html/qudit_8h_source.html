35  // by default we have a standard qubit in state |0>
36  Qudit(const cmat &rho = States::get_instance().pz0) :
37  _rho(cmat::Zero(2, 2)), _D(2) // qubit by default
38  {
39  if (!internal::_check_nonzero_size(rho))
40  throw Exception("qpp::experimental::Qudit::Qudit()", Exception::Type::ZERO_SIZE);
41  if (!internal::_check_square_mat(rho))
42  throw Exception("qpp::experimental::Qudit::Qudit()", Exception::Type::MATRIX_NOT_SQUARE);
43  _D = rho.rows();
44 
45  _rho = rho;
46  }

48  std::size_t measure(const cmat &U, bool destructive = false)
49  {
50  if (!internal::_check_square_mat(U))
51  throw Exception("qpp::experimental::Qudit::measure()",
52  Exception::Type::MATRIX_NOT_SQUARE);
53  if (static_cast<std::size_t>(U.rows()) != _D)
54  throw Exception("qpp::experimental::Qudit::measure()", Exception::Type::DIMS_INVALID);
55 
56  std::vector<double> p(_D);
57  for (std::size_t i = 0; i < _D; ++i)
58  p[i] = std::abs((cplx) trace(prj((cmat) evects(U).col(i)) * _rho));
59 
60  std::discrete_distribution<std::size_t> dd{std::begin(p), std::end(p)};
61  std::size_t result = dd(RandomDevices::get_instance()._rng);
62 
63  if (destructive) // von Neumann
64  _rho = prj((cmat) evects(U).col(result)) * _rho
65  * prj((cmat) evects(U).col(result)) / p[result];
66 
67  return result;
68  }
69 
70  // measure in the standard basis {|j>}_{j=0}^{_D-1}
71  std::size_t measure(bool destructive = false)
72  {
73  return measure(cmat::Identity(_D, _D), destructive);
74  }
