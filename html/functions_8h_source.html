567  {

576  throw Exception("spectralpowm", Exception::Type::MATRIX_NOT_SQUARE);
577

590

951  throw Exception("grams", Exception::Type::ZERO_SIZE);
952

954  if (!internal::_check_col_vector(Vs[0]))
955  throw Exception("grams", Exception::Type::MATRIX_NOT_CVECTOR);
956 
957  // now check that all the rest match the size of the first vector
958  for (auto &&it : Vs)
959  if (it.rows() != Vs[0].rows() || it.cols() != 1)
960  throw Exception("grams", Exception::Type::DIMS_NOT_EQUAL);
961 
962  DynMat<typename Derived::Scalar> cut =
963  DynMat<typename Derived::Scalar>::Identity(Vs[0].rows(),
964  Vs[0].rows());
965 
966  DynMat<typename Derived::Scalar> vi =
967  DynMat<typename Derived::Scalar>::Zero(Vs[0].rows(), 1);
968 
969  std::vector<DynMat<typename Derived::Scalar>> outvecs;
970  // find the first non-zero vector in the list
971  std::size_t pos = 0;
972  for (pos = 0; pos < Vs.size(); ++pos)
973  {
974  if (norm(Vs[pos]) > eps) // add it as the first element
975  {
976  outvecs.push_back(Vs[pos]);
977  break;
978  }
979  }
980 
981  // start the process
982  for (std::size_t i = pos + 1; i < Vs.size(); ++i)
983  {
984  cut -= prj(outvecs[i - 1 - pos]);
985  vi = cut * Vs[i];
986  outvecs.push_back(vi);
987  }
988 
989  DynMat<typename Derived::Scalar> result(Vs[0].rows(), outvecs.size());
990 
991  std::size_t cnt = 0;
992  for (auto &&it : outvecs)
993  {
994  double normV = norm(it);
995  if (normV > eps) // we add only the non-zero vectors
996  {
997  result.col(cnt) = it / normV;
998  cnt++;
999  }
1000  }
1001  return result.block(0, 0, Vs[0].rows(), cnt);
1002  }
