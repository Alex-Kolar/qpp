140 template<typename Derived> // the G-concurrence
141 double gconcurrence(const Eigen::MatrixBase<Derived>& A)
142 {
143  const DynMat<typename Derived::Scalar> & rA = A;
144 
145  // check zero-size
146  if (!internal::_check_nonzero_size(rA))
147  throw Exception("gconcurrence", Exception::Type::ZERO_SIZE);
148  // check column vector
149  if (!internal::_check_col_vector(rA))
150  throw Exception("gconcurrence", Exception::Type::MATRIX_NOT_CVECTOR);
151 
152  std::size_t D = static_cast<std::size_t>(std::sqrt((double)A.rows()));
153  if (D * D != static_cast<std::size_t>(A.rows()))
154  throw Exception("gconcurrence", Exception::Type::DIMS_NOT_EQUAL);
155 
156  // we compute exp(logdet()) to avoid underflow
157  return D * std::abs(std::exp(2. / D * logdet(reshape(rA, D, D))));
158 }
