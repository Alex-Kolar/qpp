71

101 {
102  // EXCEPTION CHECKS
103  if (!internal::_check_nonzero_size(Ks))
104  throw Exception("choi", Exception::Type::ZERO_SIZE);
105  if (!internal::_check_nonzero_size(Ks[0]))
106  throw Exception("choi", Exception::Type::ZERO_SIZE);
107  if (!internal::_check_square_mat(Ks[0]))
108  throw Exception("choi", Exception::Type::MATRIX_NOT_SQUARE);
109  for (auto it : Ks)
110  if (it.rows() != Ks[0].rows() || it.cols() != Ks[0].rows())
111  throw Exception("choi", Exception::Type::DIMS_NOT_EQUAL);
112  std::size_t D = static_cast<std::size_t>(Ks[0].rows());
113 
114  // construct the D x D \sum |jj> vector
115  // (un-normalized maximally entangled state)
116  cmat MES = cmat::Zero(D * D, 1);
117  for (std::size_t a = 0; a < D; a++)
118  MES(a * D + a) = 1;
119 
120  cmat Omega = static_cast<cmat>(MES * adjoint(MES));
121 
122  cmat result = cmat::Zero(D * D, D * D);
123  for (auto it : Ks)
124  result += kron(cmat::Identity(D, D), it) * Omega
125  * adjoint(kron(cmat::Identity(D, D), it));
126 
127  return result;
128 }

143 std::vector<cmat> choi2kraus(const cmat& A)
144 {
145  // EXCEPTION CHECKS
146  if (!internal::_check_nonzero_size(A))
147  throw Exception("choi2kraus", Exception::Type::ZERO_SIZE);
148  if (!internal::_check_square_mat(A))
149  throw Exception("choi2kraus", Exception::Type::MATRIX_NOT_SQUARE);
150  std::size_t D = static_cast<std::size_t>(std::sqrt((double)A.rows()));
151  if (D * D != static_cast<std::size_t>(A.rows()))
152  throw Exception("choi2kraus", Exception::Type::DIMS_INVALID);
153 
154  dmat ev = hevals(A);
155  cmat evec = hevects(A);
156  std::vector<cmat> result;
157 
158  for (std::size_t i = 0; i < D * D; i++)
159  {
160  // take the absolute value to get rid of tiny negatives
161  if (std::abs((double) ev(i)) > eps)
162  result.push_back(
163  (cmat) (std::sqrt((double) ev(i))
164  * reshape((cmat) evec.col(i), D, D)));
165  }
166  return result;
167 }

179 cmat channel(const Eigen::MatrixBase<Derived>& rho,
180  const std::vector<cmat>& Ks)
181 {
182  const cmat & rrho = rho;
183 
184  // EXCEPTION CHECKS
185  if (!internal::_check_nonzero_size(rrho))
186  throw Exception("channel", Exception::Type::ZERO_SIZE);
187  if (!internal::_check_square_mat(rrho))
188  throw Exception("channel", Exception::Type::MATRIX_NOT_SQUARE);
189  if (!internal::_check_nonzero_size(Ks))
190  throw Exception("channel", Exception::Type::ZERO_SIZE);
191  if (!internal::_check_square_mat(Ks[0]))
192  throw Exception("channel", Exception::Type::MATRIX_NOT_SQUARE);
193  if (Ks[0].rows() != rrho.rows())
194  throw Exception("channel", Exception::Type::DIMS_MISMATCH_MATRIX);
195  for (auto it : Ks)
196  if (it.rows() != Ks[0].rows() || it.cols() != Ks[0].rows())
197  throw Exception("channel", Exception::Type::DIMS_NOT_EQUAL);
198 
199  cmat result = cmat::Zero(rrho.rows(), rrho.cols());
200  for (auto it : Ks)
201  result += it * rrho * adjoint(it);
202 
203  return result;
204 }

248  throw Exception("channel", Exception::Type::ZERO_SIZE);

250  throw Exception("channel", Exception::Type::MATRIX_NOT_SQUARE);

323  // get the subsys' row multi-index

335  // get the subsys' col multi-index
336  for (std::size_t k = 0; k < nA; k++)
337  midxA_col[k] = midx_col[CsubsysA[k]];
338 
339  // now compute the coefficient

341  for (std::size_t a = 0; a < DA; a++)
342  {

344  internal::_n2multiidx(a, nA, CdimsA, midxA_rho_row);

354  midx_rho_col[CsubsysA[k]] = midxA_rho_col[k];

358  midx_rho_row[CsubsysA_bar[k]] = midxA_bar_row[k];
359  midx_rho_col[CsubsysA_bar[k]] = midxA_bar_col[k];
360  }
361

363  std::size_t midx_sop_row[2];
364  std::size_t sop_dims[2];
365  sop_dims[0] = sop_dims[1] = DA;

367  CdimsA);
368  midx_sop_row[1] = internal::_multiidx2n(midxA_col, nA,
369  CdimsA);

371  midx_sop_col[1] = b;

381  }

384  }
385  }
