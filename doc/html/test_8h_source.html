51 template<typename Derived>
52 std::tuple<std::vector<idx>, double, cmat>
53 measure_seq(
54  const Eigen::MatrixBase<Derived>& A,
55  std::vector<idx> subsys,
56  std::vector<idx> dims)
57 {
58  dyn_mat<typename Derived::Scalar> cA = A;
59 
60  // EXCEPTION CHECKS
61 
62  // check zero-size
63  if (!internal::_check_nonzero_size(cA))
64  throw Exception("qpp::measure_seq()", Exception::Type::ZERO_SIZE);
65 
66  // check that dimension is valid
67  if (!internal::_check_dims(dims))
68  throw Exception("qpp::measure_seq()", Exception::Type::DIMS_INVALID);
69 
70  // check that dims match rho matrix
71  if (!internal::_check_dims_match_mat(dims, cA))
72  throw Exception("qpp::measure_seq()",
73  Exception::Type::DIMS_MISMATCH_MATRIX);
74 
75  // check subsys is valid w.r.t. dims
76  if (!internal::_check_subsys_match_dims(subsys, dims))
77  throw Exception("qpp::measure_seq()",
78  Exception::Type::SUBSYS_MISMATCH_DIMS);
79 
80  // END EXCEPTION CHECKS
81 
82  std::vector<idx> result;
83  double prob = 1;
84 
85  // sort subsys in decreasing order,
86  // the order of measurements does not matter
87  std::sort(std::begin(subsys), std::end(subsys), std::greater<idx>{});
88 
89  //************ density matrix or column vector ************//
90  if (internal::_check_square_mat(cA) || internal::_check_cvector(cA))
91  {
92  while (subsys.size() > 0)
93  {
94  auto tmp = measure(cA, gt.Id(dims.at(subsys.at(0))),
95  {subsys.at(0)}, dims);
96  result.push_back(std::get<0>(tmp));
97  prob *= std::get<1>(tmp).at(std::get<0>(tmp));
98  cA = std::get<2>(tmp).at(std::get<0>(tmp));
99 
100  // remove the subsystem
101  dims.erase(std::next(dims.begin(), subsys.at(0)));
102  subsys.erase(subsys.begin());
103  }
104  }
105  else
106  throw Exception("qpp::measure_seq()",
107  Exception::Type::MATRIX_NOT_SQUARE_OR_CVECTOR);
108 
109  return std::make_tuple(result, prob, cA);
110 }
