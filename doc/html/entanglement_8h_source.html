40 template<typename Derived>
41 DynColVect<cplx> schmidtcoeff(const Eigen::MatrixBase<Derived>& A,
42  const std::vector<std::size_t>& dims)
43 {
44  const DynMat<typename Derived::Scalar>& rA = A;
45  // check zero-size
46  if (!internal::_check_nonzero_size(rA))
47  throw Exception("qpp::schmidtcoeff()", Exception::Type::ZERO_SIZE);
48  // check bi-partite
49  if (dims.size() != 2)
50  throw Exception("qpp::schmidtcoeff()", Exception::Type::NOT_BIPARTITE);
51  // check column vector
52  if (!internal::_check_col_vector(rA))
53  throw Exception("qpp::schmidtcoeff()",
54  Exception::Type::MATRIX_NOT_CVECTOR);
55  // check matching dimensions
56  if (!internal::_check_dims_match_mat(dims, rA))
57  throw Exception("qpp::schmidtcoeff()",
58  Exception::Type::DIMS_MISMATCH_MATRIX);
59 
60  Eigen::JacobiSVD<DynMat<typename Derived::Scalar>> svd(
61  transpose(reshape(rA, dims[1], dims[0])));
62 
63  return svd.singularValues().template cast<cplx>();
64 }

213 template<typename Derived>
214 // the G-concurrence
215 double gconcurrence(const Eigen::MatrixBase<Derived>& A)
216 {
217  const DynMat<typename Derived::Scalar>& rA = A;
218 
219  // check zero-size
220  if (!internal::_check_nonzero_size(rA))
221  throw Exception("qpp::gconcurrence()", Exception::Type::ZERO_SIZE);
222  // check column vector
223  if (!internal::_check_col_vector(rA))
224  throw Exception("qpp::gconcurrence()",
225  Exception::Type::MATRIX_NOT_CVECTOR);
226 
227  // check equal local dimensions
228  std::size_t D = static_cast<std::size_t>(std::llround(
229  std::sqrt(static_cast<double>(rA.rows()))));
230  if (D * D != static_cast<std::size_t>(rA.rows()))
231  throw Exception("qpp::gconcurrence()",
232  Exception::Type::DIMS_NOT_EQUAL);
233 
234  // we compute exp(logdet()) to avoid underflow
235  return D * std::abs(std::exp(2. / D * logdet(reshape(rA, D, D))));
236 }

245 template<typename Derived>

247  const std::vector<std::size_t>& dims)
248 {

253  // check bi-partite
254  if (dims.size() != 2)
255  throw Exception("qpp::negativity()", Exception::Type::NOT_BIPARTITE);
256  // check square matrix vector
257  if (!internal::_check_square_mat(rA))
258  throw Exception("qpp::negativity()",
259  Exception::Type::MATRIX_NOT_SQUARE);
260  // check matching dimensions
261  if (!internal::_check_dims_match_mat(dims, rA))
262  throw Exception("qpp::negativity()",
263  Exception::Type::DIMS_MISMATCH_MATRIX);
264 
265  return (schatten(ptranspose(rA, {0}, dims), 1) - 1.) / 2.;
266 }

305 template<typename Derived>
306 double concurrence(const Eigen::MatrixBase<Derived>& A)
307 {
308  const DynMat<typename Derived::Scalar>& rA = A;
309  // check zero-size
310  if (!internal::_check_nonzero_size(rA))
311  throw Exception("qpp::concurrence()", Exception::Type::ZERO_SIZE);
312  // check square matrix vector
313  if (!internal::_check_square_mat(rA))
314  throw Exception("qpp::concurrence()",
315  Exception::Type::MATRIX_NOT_SQUARE);
316  // check that the state is a 2-qubit state
317  if (rA.rows() != 4)
318  throw Exception("qpp::concurrence()",
319  Exception::Type::NOT_QUBIT_SUBSYS);
320 
321  cmat sigmaY = Gates::get_instance().Y;
322  DynColVect<double> lambdas =
323  evals(
324  rA * kron(sigmaY, sigmaY) * conjugate(rA)
325  * kron(sigmaY, sigmaY)).real();
326 
327  std::vector<double> lambdas_sorted(lambdas.data(),
328  lambdas.data() + lambdas.size());
329 
330  std::sort(std::begin(lambdas_sorted), std::end(lambdas_sorted),
331  std::greater<double>());
332  std::transform(std::begin(lambdas_sorted), std::end(lambdas_sorted),
333  std::begin(lambdas_sorted), [](double elem)
334  {
335  return std::sqrt(std::abs(elem));
336  }); // chop tiny negatives
337 
338  return std::max(0.,
339  lambdas_sorted[0] - lambdas_sorted[1] - lambdas_sorted[2]
340  - lambdas_sorted[3]);
341 }
