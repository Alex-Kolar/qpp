507  QCircuitDescription& gate(const cmat& U, idx i, std::string name = "") {
508  // EXCEPTION CHECKS
509 
510  try {
511  // check valid target
512  if (i > nq_)
513  throw exception::OutOfRange("qpp::QCircuitDescription::gate()");
514  // check not measured before
515  if (get_measured(i))
516  throw exception::QuditAlreadyMeasured(
517  "qpp::QCircuitDescription::gate()");
518 
519  // check square matrix for the gate
520  if (!internal::check_square_mat(U))
521  throw exception::MatrixNotSquare(
522  "qpp::QCircuitDescription::gate()");
523  // check correct dimension
524  if (static_cast<idx>(U.rows()) != d_)
525  throw exception::DimsMismatchMatrix(
526  "qpp::QCircuitDescription::gate()");
527  } catch (qpp::exception::Exception& e) {
528  std::cerr << "At STEP " << step_cnt_ << "\n";
529  throw;
530  }
531  // END EXCEPTION CHECKS
532 
533  if (name == "")
534  name = qpp::Gates::get_instance().get_name(U);
535  gates_.emplace_back(GateType::SINGLE, U, std::vector<idx>{},
536  std::vector<idx>{i}, step_cnt_++, name);
537 
538  return *this;
539  }
