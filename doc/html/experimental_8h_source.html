377  std::ostream& display(std::ostream& os) const override {
378  os << "nq = " << nq_ << ", nc = " << nc_ << ", d = " << d_;
379 
380  if (name_ != "") // if the circuit is named
381  os << ", name = \"" << name_ << "\"\n";
382  else
383  os << ", name = \"\"\n";
384 
385  idx gates_size = gates_.size();
386  idx measurements_size = measurement_steps_.size();
387 
388  idx measurement_ip = 0; // measurement instruction pointer
389  for (idx i = 0; i <= gates_size; ++i) {
390  // check for measurements before gates
391  if (measurements_size != 0) {
392  idx current_measurement_step =
393  measurement_steps_[measurement_ip];
394  // we have a measurement
395  if (current_measurement_step == i) {
396  while (measurement_steps_[measurement_ip] ==
397  current_measurement_step) {
398  os << measurements_[measurement_ip].measurement_type_
399  << ", "
400  << "\"" << measurements_[measurement_ip].name_
401  << "\""
402  << ", ";
403  os << disp(measurements_[measurement_ip].target_, ",");
404  os << '\n';
405  ++measurement_ip;
406  }
407  }
408  }
409  // check for gates
410  if (i < gates_size) {
411  os << gates_[i].gate_type_ << ", "
412  << "\"" << gates_[i].name_ << "\""
413  << ", ";
414  if (gates_[i].gate_type_ >=
415  GateType ::SINGLE_CTRL_SINGLE_TARGET)
416  os << disp(gates_[i].ctrl_, ",") << ", ";
417  os << disp(gates_[i].target_, ",");
418  os << '\n';
419  }
420  }
421 
422  os << "measurement steps (assumed after gate steps): ";
423  std::cout << disp(measurement_steps_, ",") << '\n';
424 
425  return os;
426  }
