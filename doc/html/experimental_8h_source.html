49 template<typename Scalar>
50 dyn_col_vect <Scalar> ip(
51  const dyn_col_vect <Scalar>& phi,
52  const dyn_col_vect <Scalar>& psi,
53  const std::vector<idx>& subsys,
54  const std::vector<idx>& dims)
55 {
56  const dyn_col_vect<Scalar>& rphi = phi;
57  const dyn_col_vect<Scalar>& rpsi = psi;
58 
59  // EXCEPTION CHECKS TODO
60  if (!internal::_check_dims_match_cvect(dims, rpsi))
61  throw Exception("qpp::ip()",
62  Exception::Type::DIMS_MISMATCH_CVECTOR);
63 
64  std::vector<idx> subsys_dims(subsys.size());
65  for (idx i = 0; i < subsys.size(); ++i)
66  subsys_dims[i] = dims[subsys[i]];
67 
68  idx Dsubsys = prod(std::begin(subsys_dims), std::end(subsys_dims));
69 
70  idx D = static_cast<idx>(rpsi.rows());
71  idx Dbar = D / Dsubsys;
72 
73  idx n = dims.size();
74  idx nsubsys = subsys.size();
75  idx nsubsysbar = n - nsubsys;
76 
77  idx Cdims[maxn];
78  idx Csubsys[maxn];
79  idx Cdimssubsys[maxn];
80  idx Csubsysbar[maxn];
81  idx Cdimssubsysbar[maxn];
82 
83  std::vector<idx> subsys_bar = complement(subsys, n);
84  std::copy(std::begin(subsys_bar), std::end(subsys_bar),
85  std::begin(Csubsysbar));
86 
87  for (idx i = 0; i < n; ++i)
88  {
89  Cdims[i] = dims[i];
90  }
91  for (idx i = 0; i < nsubsys; ++i)
92  {
93  Csubsys[i] = subsys[i];
94  Cdimssubsys[i] = dims[subsys[i]];
95  }
96  for (idx i = 0; i < nsubsysbar; ++i)
97  {
98  Cdimssubsysbar[i] = dims[subsys_bar[i]];
99  }
100 
101  auto worker = [=](idx b) noexcept
102  -> Scalar
103  {
104  idx Cmidxrow[maxn];
105  idx Cmidxrowsubsys[maxn];
106  idx Cmidxcolsubsysbar[maxn];
107 
108  /* get the col multi-indexes of the complement */
109  internal::_n2multiidx(b, nsubsysbar,
110  Cdimssubsysbar, Cmidxcolsubsysbar);
111  /* write it in the global row multi-index */
112  for (idx k = 0; k < nsubsysbar; ++k)
113  {
114  Cmidxrow[Csubsysbar[k]] = Cmidxcolsubsysbar[k];
115  }
116 
117  Scalar result = 0;
118 
119  for (idx a = 0; a < Dsubsys; ++a)
120  {
121  /* get the row multi-indexes of the subsys */
122  internal::_n2multiidx(a, nsubsys,
123  Cdimssubsys, Cmidxrowsubsys);
124  /* write it in the global row multi-index */
125  for (idx k = 0; k < nsubsys; ++k)
126  {
127  Cmidxrow[Csubsys[k]] = Cmidxrowsubsys[k];
128  }
129  // compute the row index
130  idx i = internal::_multiidx2n(Cmidxrow, n, Cdims);
131 
132  // TODO: check whether need adjoint(V.col(m)(a))
133  result += std::conj(rphi(a)) * rpsi(i);
134  }
135 
136  return result;
137  }; /* end worker */
138 
139  dyn_col_vect<Scalar> result(Dbar);
140 #pragma omp parallel for
141  for (idx m = 0; m < Dbar; ++m)
142  result(m) = worker(m);
143 
144  return result;
145 }

197 _measure(const Eigen::MatrixBase<Derived>& A,
198  const cmat& V,
199  const std::vector<idx>& subsys,
200  const std::vector<idx>& dims)
201 {
202  const cmat& rA = A;
203 
204  // EXCEPTION CHECKS
205 
206  // check zero-size
207  if (!internal::_check_nonzero_size(rA))
208  throw Exception("qpp::_measure()", Exception::Type::ZERO_SIZE);
209 
210  // check that dimension is valid
211  if (!internal::_check_dims(dims))
212  throw Exception("qpp::_measure()", Exception::Type::DIMS_INVALID);
213 
214  // check subsys is valid w.r.t. dims
215  if (!internal::_check_subsys_match_dims(subsys, dims))
216  throw Exception("qpp::_measure()",
217  Exception::Type::SUBSYS_MISMATCH_DIMS);
218 
219  std::vector<idx> subsys_dims(subsys.size());
220  for (idx i = 0; i < subsys.size(); ++i)
221  subsys_dims[i] = dims[subsys[i]];
222 
223  idx Dsubsys = prod(std::begin(subsys_dims), std::end(subsys_dims));
224 
225  // check the matrix V
226  if (!internal::_check_nonzero_size(V))
227  throw Exception("qpp::_measure()", Exception::Type::ZERO_SIZE);
228 // if (!internal::_check_square_mat(U))
229 // throw Exception("qpp::measure()", Exception::Type::MATRIX_NOT_SQUARE);
230  if (Dsubsys != static_cast<idx>(V.rows()))
231  throw Exception("qpp::_measure()",
232  Exception::Type::DIMS_MISMATCH_MATRIX);
233  // END EXCEPTION CHECKS
234 
235  // number of basis (rank-1 POVM) elements
236  idx M = static_cast<idx>(V.cols());
237 
238  // TODO: modify this!!! (and document U in case of rank-one POVMs)
239  //************ ket ************//
240  if (internal::_check_cvector(rA))
241  {
242  PRINTLN("Ket");
243  const ket& rpsi = A;
244  // check that dims match state vector
245  if (!internal::_check_dims_match_cvect(dims, rA))
246  throw Exception("qpp::_measure()",
247  Exception::Type::DIMS_MISMATCH_CVECTOR);
248 
249  std::vector<double> prob(M); // probabilities
250  std::vector<cmat> outstates(M); // resulting states
251 
252 #pragma omp parallel for
253  for (idx i = 0; i < M; ++i)
254  outstates[i] = ip(static_cast<const ket&>(V.col(i)),
255  rpsi, subsys, dims);
256 
257  for (idx i = 0; i < M; ++i)
258  {
259  double tmp = norm(outstates[i]);
260  prob[i] = tmp * tmp;
261  if (prob[i] > eps)
262  {
263  // normalized output state
264  // corresponding to measurement result m
265  outstates[i] /= tmp;
266  }
267  }
268 
269  // sample from the probability distribution
270  std::discrete_distribution<idx> dd(std::begin(prob),
271  std::end(prob));
272  idx result = dd(RandomDevices::get_instance()._rng);
273 
274  return std::make_tuple(result, prob, outstates);
275  }
276  //************ density matrix ************//
277  else if (internal::_check_square_mat(rA))
278  {
279  PRINTLN("Matrix");
280  // check that dims match rho matrix
281  if (!internal::_check_dims_match_mat(dims, rA))
282  throw Exception("qpp::_measure()",
283  Exception::Type::DIMS_MISMATCH_MATRIX);
284 
285  std::vector<cmat> Ks(M);
286  for (idx i = 0; i < M; ++i)
287  Ks[i] = V.col(i) * adjoint(V.col(i));
288 
289  return measure(rA, Ks, subsys, dims);
290  }
291  //************ Exception: not ket nor density matrix ************//
292  throw Exception("qpp::_measure()",
293  Exception::Type::MATRIX_NOT_SQUARE_OR_CVECTOR);
294 }

350 template<typename Derived>
351 std::tuple<std::vector<idx>, double, cmat>
352 _measure_seq(const Eigen::MatrixBase<Derived>& A,
353  std::vector<idx> subsys,
354  std::vector<idx> dims)
355 {
356  dyn_mat<typename Derived::Scalar> cA = A;
357 
358  // EXCEPTION CHECKS
359 
360  // check zero-size
361  if (!internal::_check_nonzero_size(cA))
362  throw Exception("qpp::measure_seq()", Exception::Type::ZERO_SIZE);
363 
364  // check that dimension is valid
365  if (!internal::_check_dims(dims))
366  throw Exception("qpp::measure_seq()", Exception::Type::DIMS_INVALID);
367 
368  // check that dims match rho matrix
369  if (!internal::_check_dims_match_mat(dims, cA))
370  throw Exception("qpp::measure_seq()",
371  Exception::Type::DIMS_MISMATCH_MATRIX);
372 
373  // check subsys is valid w.r.t. dims
374  if (!internal::_check_subsys_match_dims(subsys, dims))
375  throw Exception("qpp::measure_seq()",
376  Exception::Type::SUBSYS_MISMATCH_DIMS);
377 
378  // END EXCEPTION CHECKS
379 
380  std::vector<idx> result;
381  double prob = 1;
382 
383  // sort subsys in decreasing order,
384  // the order of measurements does not matter
385  std::sort(std::begin(subsys), std::end(subsys), std::greater<idx>{});
386 
387  // TODO: modify this so it treats both density matrices and column vectors
388  //************ density matrix or column vector ************//
389  if (internal::_check_square_mat(cA) || internal::_check_cvector(cA))
390  {
391  while (subsys.size() > 0)
392  {
393  auto tmp = experimental::_measure(
394  cA, Gates::get_instance().Id(dims[subsys[0]]),
395  {subsys[0]}, dims
396  );
397  result.push_back(std::get<0>(tmp));
398  prob *= std::get<1>(tmp)[std::get<0>(tmp)];
399  cA = std::get<2>(tmp)[std::get<0>(tmp)];
400 
401  // remove the subsystem
402  dims.erase(std::next(std::begin(dims), subsys[0]));
403  subsys.erase(std::begin(subsys));
404  }
405  // order result in increasing order with respect to subsys
406  std::reverse(std::begin(result), std::end(result));
407  }
408  else
409  throw Exception("qpp::_measure_seq()",
410  Exception::Type::MATRIX_NOT_SQUARE_OR_CVECTOR);
411 
412  return std::make_tuple(result, prob, cA);
413 }
