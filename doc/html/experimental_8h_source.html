130  // count the bits set to true
131  // (i.e. computes the Hamming distance)
136  idx count() const noexcept
137  {
138  std::size_t result = 0;
139  for (idx i = 0; i < size(); ++i)
140  {
141  if (this->get(i))
142  ++result;
143  }
144 
145  return result;
146  }

148  // returns the bit at position pos
154  bool get(idx pos) const
155  {
156  return 1 & (v_[index_(pos)] >> offset_(pos));
157  }
158 
159  // returns true if none of the bits are set
164  bool none() const noexcept
165  {
166  bool result = true;
167  for (idx i = 0; i < storage_size(); ++i)
168  {
169  if (v_[i])
170  {
171  return false;
172  }
173  }
174 
175  return result;
176  }
177 
178  // returns true if all bits are set to true
183  bool all() const noexcept
184  {
185  bool result = true;
186  for (idx i = 0; i < storage_size(); ++i)
187  {
188  if (~v_[i])
189  {
190  return false;
191  }
192  }
193 
194  return result;
195  }
196 
197  // returns true if any bit is set to true
202  bool any() const noexcept
203  {
204  return !(this->none());
205  }

207  /* setters */
208  // set the bit to a specific value
215  Dynamic_bitset& set(idx pos, bool value = true)
216  {
217  value ? v_[index_(pos)] |=
218  (1 << offset_(pos)) :
219  v_[index_(pos)] &= ~(1 << offset_(pos));
220 
221 // v_[index_(pos)] &= ~(!value << offset_(pos));
222 
223  return *this;

310  {

316  // flip all bits
321  Dynamic_bitset& flip() noexcept
322  {

342  for (idx i = 0; i < n; ++i)

403  {

442 
443  // flips the bit
444  Bit_circuit& X(idx pos)
445  {
446  this->flip(pos);
447  ++gate_count.X;
448 
449  return *this;
450  }
451 
452  // flips the bit
453  Bit_circuit& NOT(idx pos)
454  {
455  this->flip(pos);
456  ++gate_count.NOT;
457 
458  return *this;
459  }
460 
461  // controlled-NOT control-target
462  Bit_circuit& CNOT(const std::vector<idx>& pos)
463  {
464  v_[index_(pos[1])] ^=
465  (1 & (v_[index_(pos[0])] >> offset_(pos[0])))
466  << offset_(pos[1]);
467  ++gate_count.CNOT;
468 
469  return *this;
470  }
471 
472  // Toffoli control-control-target
473  Bit_circuit& TOF(const std::vector<idx>& pos)
474  {
475  v_[index_(pos[2])] ^=
476  ((1 & (v_[index_(pos[1])] >> offset_(pos[1])))
477  &
478  (1 & (v_[index_(pos[0])] >> offset_(pos[0])))
479  ) << offset_(pos[2]);
480 
481  ++gate_count.TOF;
482  return *this;
483  }
484 
485  // SWAP 2 bits
486  Bit_circuit& SWAP(const std::vector<idx>& pos)
487  {
488  if (this->get(pos[0]) != this->get(pos[1]))
489  {
490  this->X(pos[0]);
491  this->X(pos[1]);
492  }
493 
494  ++gate_count.SWAP;
495  return *this;
496  }
497 
498  // Fredkin gate
499  Bit_circuit& FRED(const std::vector<idx>& pos)
500  {
501  if (this->get(pos[0]))
502  {
503  this->SWAP({pos[1], pos[2]});
504  }
505 
506  ++gate_count.FRED;
507  return *this;
508  }
509 
510  // reset the circuit all zero, clear all gates
511  Bit_circuit& reset() noexcept
512  {
513  gate_count.NOT = gate_count.X = 0;
514  gate_count.CNOT = gate_count.SWAP = 0;
515  gate_count.FRED = gate_count.TOF = 0;
516 
517  return *this;
518  }
