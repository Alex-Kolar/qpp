557 class QCircuit : public IDisplay {
558  QCircuitDescription qcd_;
559  ket psi_;
560  std::vector<idx> dits_;
561  std::vector<double> probs_;
562  std::vector<idx> subsys_;
563  std::vector<bool> measured_;
565 
566  idx was_measured(idx i) const { return measured_[i]; }
567 
568  // mark qudit i as measured then re-label accordingly the remaining
569  // non-measured qudits
570  void mark_as_measured_(idx i) {
571  if (was_measured(i))
572  throw qpp::exception::QuditAlreadyMeasured(
573  "qpp::QCircuit::mark_as_measured_()");
574  subsys_[i] = idx_infty; // set qudit i to measured state
575  for (idx m = i; m < qcd_.nq_; ++m) {
576  if (!was_measured(m)) {
577  --subsys_[m];
578  }
579  }
580  }
581 
582  // giving a vector of non-measured qudits, get their relative position wrt
583  // the measured qudits
584  std::vector<idx> get_relative_pos_(std::vector<idx> v) {
585  idx vsize = v.size();
586  for (idx i = 0; i < vsize; ++i) {
587  if (was_measured(v[i]))
588  throw qpp::exception::QuditAlreadyMeasured(
589  "qpp::QCircuit::get_relative_pos_()");
590  // if (v[i] >= nq_)
591  // throw qpp::exception::SubsysMismatchDims(
592  // "qpp::QCircuit::get_relative_pos_()");
593  v[i] = subsys_[v[i]];
594  }
595  return v;
596  }

601  probs_(qcd.nc_, 0), subsys_(qcd.nq_, 0), measured_(qcd.nq_, false) {
602  std::iota(std::begin(subsys_), std::end(subsys_), 0);
603  }

624  void run() {
625  idx gates_size_ = qcd_.gates_.size();
626  idx measurements_size_ = qcd_.measurements_.size();
627  idx m_ip_ = 0; // measurement instruction pointer
628 
629  for (idx i = 0; i <= gates_size_; ++i) {
630 
631  // check for measurements
632  if (m_ip_ < measurements_size_) {
633  idx m_step = qcd_.measurement_steps_[m_ip_];
634  // we have a measurement at step i
635  if (m_step == i) {
636  while (qcd_.measurement_steps_[m_ip_] == m_step) {
637  std::vector<idx> target_rel_pos =
638  get_relative_pos_(qcd_.measurements_[m_ip_].target_);
639 
640  std::vector<idx> resZ;
641  double probZ;
642 
643  idx mres{};
644  std::vector<double> probs;
645  std::vector<cmat> states_;
646 
647  switch (qcd_.measurements_[m_ip_].measurement_type_) {
648  case QCircuitDescription::MeasureType::NONE:
649  break;
650  case QCircuitDescription::MeasureType::MEASURE_Z:
651  std::tie(resZ, probZ, psi_) =
652  qpp::measure_seq(psi_, target_rel_pos, qcd_.d_);
653  dits_[qcd_.measurements_[m_ip_].c_reg_] = resZ[0];
654  probs_[qcd_.measurements_[m_ip_].c_reg_] = probZ;
655  mark_as_measured_(
656  qcd_.measurements_[m_ip_].target_[0]);
657  break;
658  case QCircuitDescription::MeasureType::MEASURE_V:
659  std::tie(mres, probs, states_) = qpp::measure(
660  psi_, qcd_.measurements_[m_ip_].mats_[0],
661  target_rel_pos, qcd_.d_);
662  std::cout << disp(states_[mres]) << "\n";
663  psi_ = states_[mres];
664  dits_[qcd_.measurements_[m_ip_].c_reg_] = mres;
665  probs_[qcd_.measurements_[m_ip_].c_reg_] =
666  probs[mres];
667  mark_as_measured_(
668  qcd_.measurements_[m_ip_].target_[0]);
669  break;
670  }
671 
672  if (++m_ip_ == measurements_size_)
673  break;
674  }
675  }
676  }
677 
678  // check for gates
679  if (i < gates_size_) {
680  std::vector<idx> ctrl_rel_pos =
681  get_relative_pos_(qcd_.gates_[i].ctrl_);
682  std::vector<idx> target_rel_pos =
683  get_relative_pos_(qcd_.gates_[i].target_);
684 
685  switch (qcd_.gates_[i].gate_type_) {
686  case QCircuitDescription::GateType::NONE:
687  break;
688  case QCircuitDescription::GateType::SINGLE:
