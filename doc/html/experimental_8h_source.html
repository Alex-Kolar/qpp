125  // count the bits set to true
126  // (i.e. computes the Hamming distance)
131  idx count() const noexcept
132  {
133  std::size_t result = 0;
134  for (idx i = 0; i < size(); ++i)
135  {
136  if (this->get(i))
137  ++result;
138  }
139 
140  return result;
141  }

143  // returns the bit at position pos
149  bool get(idx pos) const
150  {
151  return 1 & (v_[index_(pos)] >> offset_(pos));
152  }
153 
154  // returns true if none of the bits are set
159  bool none() const noexcept
160  {
161  bool result = true;
162  for (idx i = 0; i < storage_size(); ++i)
163  {
164  if (v_[i])
165  {
166  return false;
167  }
168  }
169 
170  return result;
171  }
172 
173  // returns true if all bits are set to true
178  bool all() const noexcept
179  {
180  bool result = true;
181  for (idx i = 0; i < storage_size(); ++i)
182  {
183  if (~v_[i])
184  {
185  return false;
186  }
187  }
188 
189  return result;
190  }
191 
192  // returns true if any bit is set to true
197  bool any() const noexcept
198  {
199  return !(this->none());
200  }

202  /* setters */
203  // set the bit to a specific value
210  Dynamic_bitset& set(idx pos, bool value = true)
211  {
212  value ? v_[index_(pos)] |=
213  (1 << offset_(pos)) :
214  v_[index_(pos)] &= ~(1 << offset_(pos));
215 
216 // v_[index_(pos)] &= ~(!value << offset_(pos));
217 
218  return *this;

316  Dynamic_bitset& flip() noexcept

332  bool operator==(const Dynamic_bitset& rhs) const noexcept
333  {
334  assert(this->size() == rhs.size());
335  bool result = true;
336  idx n = std::min(this->storage_size(), rhs.storage_size());
337  for (idx i = 0; i < n; ++i)
338  {
339  if (v_[i] != rhs.v_[i])
340  {
341  return false;
342  }
343  }
344 
345  return result;
346  }

392  {

395  result.resize(bitset_size);
396 
397  for (idx i = bitset_size; i-- > 0;)
398  {
399  if (!this->get(i))
400  {
401  result[bitset_size - i - 1] = zero;
402  } else
403  {
404  result[bitset_size - i - 1] = one;

438  // flips the bit
439  Bit_circuit& X(idx pos)
440  {
441  this->flip(pos);
442  ++gate_count.X;
443 
444  return *this;
445  }
446 
447  // flips the bit
448  Bit_circuit& NOT(idx pos)
449  {
450  this->flip(pos);
451  ++gate_count.NOT;
452 
453  return *this;
454  }
455 
456  // controlled-NOT control-target
457  Bit_circuit& CNOT(const std::vector<idx>& pos)
458  {
459  v_[index_(pos[1])] ^=
460  (1 & (v_[index_(pos[0])] >> offset_(pos[0])))
461  << offset_(pos[1]);
462  ++gate_count.CNOT;
463 
464  return *this;
465  }
466 
467  // Toffoli control-control-target
468  Bit_circuit& TOF(const std::vector<idx>& pos)
469  {
470  v_[index_(pos[2])] ^=
471  ((1 & (v_[index_(pos[1])] >> offset_(pos[1])))
472  &
473  (1 & (v_[index_(pos[0])] >> offset_(pos[0])))
474  ) << offset_(pos[2]);
475 
476  ++gate_count.TOF;
477  return *this;
478  }
479 
480  // SWAP 2 bits
481  Bit_circuit& SWAP(const std::vector<idx>& pos)
482  {
483  if (this->get(pos[0]) != this->get(pos[1]))
484  {
485  this->X(pos[0]);
486  this->X(pos[1]);
487  }
488 
489  ++gate_count.SWAP;
490  return *this;
491  }
492 
493  // Fredkin gate
494  Bit_circuit& FRED(const std::vector<idx>& pos)
495  {
496  if (this->get(pos[0]))
497  {
498  this->SWAP({pos[1], pos[2]});
499  }
500 
501  ++gate_count.FRED;
502  return *this;
503  }
504 
505  // reset the circuit all zero, clear all gates
506  Bit_circuit& reset() noexcept
507  {
508  gate_count.NOT = gate_count.X = 0;
509  gate_count.CNOT = gate_count.SWAP = 0;
510  gate_count.FRED = gate_count.TOF = 0;
511 
512  return *this;
513  }
