39 template<typename Derived>
40 double shannon(const Eigen::MatrixBase <Derived>& A)
41 {
42  const dyn_mat<typename Derived::Scalar>& rA = A;
43 
44  // check zero-size
45  if (!internal::_check_nonzero_size(rA))
46  throw Exception("qpp::shannon()", Exception::Type::ZERO_SIZE);
47 
48  // input is a vector
49  if (internal::_check_vector(rA))
50  {
51  double result = 0;
52  for (idx i = 0; i < static_cast<idx>(rA.rows()); ++i)
53  if (std::abs(rA(i)) != 0) // not identically zero
54  result -= std::abs(rA(i)) * std::log2(std::abs(rA(i)));
55 
56  return result;
57  }
58 
59  // input is a matrix
60 
61  // check square matrix
62  if (!internal::_check_square_mat(rA))
63  throw Exception("qpp::shannon()", Exception::Type::MATRIX_NOT_SQUARE);
64 
65  dmat ev = svals(rA); // get the singular values
66  double result = 0;
67  for (idx i = 0; i < static_cast<idx>(ev.rows()); ++i)
68  if (ev(i) != 0) // not identically zero
69  result -= ev(i) * std::log2(ev(i));
70 
71  return result;
72 }

84 template<typename Derived>
85 double renyi(const Eigen::MatrixBase <Derived>& A, double alpha)
86 {
87  const dyn_mat<typename Derived::Scalar>& rA = A;
88 
89  if (alpha < 0)
90  throw Exception("qpp::renyi()", Exception::Type::OUT_OF_RANGE);
91 
92  if (alpha == 1) // Shannon/von Neumann
93  return shannon(rA);
94 
95  // check zero-size
96  if (!internal::_check_nonzero_size(rA))
97  throw Exception("qpp::renyi()", Exception::Type::ZERO_SIZE);
98 
99  // input is a vector
100  if (internal::_check_vector(rA))
101  {
102  if (alpha == 0) // H max
103  return std::log2(static_cast<double>( rA.rows()));
104 
105  if (alpha == infty) // H min
106  {
107  double max = 0;
108  for (idx i = 0; i < static_cast<idx>(rA.rows());
109  ++i)
110  if (std::abs(rA(i)) > max)
111  max = std::abs(rA(i));
112 
113  return -std::log2(max);
114  }
115 
116  double result = 0;
117  for (idx i = 0; i < static_cast<idx>(rA.rows()); ++i)
118  if (std::abs(rA(i)) != 0) // not identically zero
119  result += std::pow(std::abs(rA(i)), alpha);
120 
121  return std::log2(result) / (1 - alpha);
122  }
123 
124  // input is a matrix
125 
126  // check square matrix
127  if (!internal::_check_square_mat(rA))
128  throw Exception("qpp::renyi()", Exception::Type::MATRIX_NOT_SQUARE);
129 
130  if (alpha == 0) // H max
131  return std::log2(static_cast<double>( rA.rows()));
132 
133  if (alpha == infty) // H min
134  return -std::log2(svals(rA)[0]);
135 
136  dmat sv = svals(rA); // get the singular values
137  double result = 0;
138  for (idx i = 0; i < static_cast<idx>(sv.rows()); ++i)
139  if (sv(i) != 0) // not identically zero
140  result += std::pow(sv(i), alpha);
141 
142  return std::log2(result) / (1 - alpha);
143 }

158 template<typename Derived>
159 double tsallis(const Eigen::MatrixBase <Derived>& A, double alpha)
160 {
161  const dyn_mat<typename Derived::Scalar>& rA = A;
162 
163  if (alpha < 0)
164  throw Exception("qpp::tsallis()", Exception::Type::OUT_OF_RANGE);
165 
166  if (alpha == 1) // Shannon/von Neumann with base e logarithm
167  return shannon(rA) * std::log(2.);
168 
169  // check zero-size
170  if (!internal::_check_nonzero_size(rA))
171  throw Exception("qpp::tsallis()", Exception::Type::ZERO_SIZE);
172 
173  // input is a vector
174  if (internal::_check_vector(rA))
175  {
176  double result = 0;
177  for (idx i = 0; i < static_cast<idx>(rA.rows()); ++i)
178  if (std::abs(rA(i)) != 0) // not identically zero
179  result += std::pow(std::abs(rA(i)), alpha);
180 
181  return (result - 1) / (1 - alpha);
182  }
183 
184  // input is a matrix
185 
186  // check square matrix
187  if (!internal::_check_square_mat(rA))
188  throw Exception("qpp::tsallis()", Exception::Type::MATRIX_NOT_SQUARE);
189 
190  dmat ev = svals(rA); // get the singular values
191  double result = 0;
192  for (idx i = 0; i < static_cast<idx>(ev.rows()); ++i)
193  if (ev(i) != 0) // not identically zero
194  result += std::pow(ev(i), alpha);
195 
196  return (result - 1) / (1 - alpha);
197 }

208 template<typename Derived>
209 double qmutualinfo(const Eigen::MatrixBase <Derived>& A,
210  const std::vector <idx>& subsysA,
211  const std::vector <idx>& subsysB,
212  const std::vector <idx>& dims)
213 {
214  const dyn_mat<typename Derived::Scalar>& rA = A;
215 
216  // error checks
217 
218  // check zero-size
219  if (!internal::_check_nonzero_size(rA))
220  throw Exception("qpp::mutualinfo()", Exception::Type::ZERO_SIZE);
221 
222  // check that dims is a valid dimension vector
223  if (!internal::_check_dims(dims))
224  throw Exception("qpp::mutualinfo()", Exception::Type::DIMS_INVALID);
225 
226  // check square matrix
227  if (!internal::_check_square_mat(rA))
228  throw Exception("qpp::mutualinfo()",
229  Exception::Type::MATRIX_NOT_SQUARE);
230 
231  // check that dims match the dimension of A
232  if (!internal::_check_dims_match_mat(dims, rA))
233  throw Exception("qpp::mutualinfo()",
234  Exception::Type::DIMS_MISMATCH_MATRIX);
235 
236  // check that subsys are valid
237  if (!internal::_check_subsys_match_dims(subsysA, dims)
238  || !internal::_check_subsys_match_dims(subsysB, dims))
239  throw Exception("qpp::mutualinfo()",
240  Exception::Type::SUBSYS_MISMATCH_DIMS);
241 
242  // The full system indexes {0,1,...,n-1}
243  std::vector <idx> full_system(dims.size());
244  std::iota(std::begin(full_system), std::end(full_system), 0);
245 
246  // Sorted input subsystems
247  std::vector <idx> subsysAsorted{subsysA};
248  std::vector <idx> subsysBsorted{subsysB};
249 
250  // sort the input subsystems (as needed by std::set_difference)
251  std::sort(std::begin(subsysAsorted), std::end(subsysAsorted));
252  std::sort(std::begin(subsysBsorted), std::end(subsysBsorted));
253 
254  // construct the complement of subsys
255  std::vector <idx> subsysAbar;
256  std::vector <idx> subsysBbar;
257  std::vector <idx> subsysABbar;
258  std::vector <idx> subsysAB;
259 
260  std::set_difference(std::begin(full_system), std::end(full_system),
261  std::begin(subsysAsorted), std::end(subsysAsorted),
262  std::back_inserter(subsysAbar));
263  std::set_difference(std::begin(full_system), std::end(full_system),
264  std::begin(subsysBsorted), std::end(subsysBsorted),
265  std::back_inserter(subsysBbar));
266  std::set_union(std::begin(subsysAsorted), std::end(subsysAsorted),
267  std::begin(subsysBsorted), std::end(subsysBsorted),
268  std::back_inserter(subsysAB));
269  std::sort(std::begin(subsysAB), std::end(subsysAB));
270 
271  std::set_difference(std::begin(full_system), std::end(full_system),
272  std::begin(subsysAB), std::end(subsysAB),
273  std::back_inserter(subsysABbar));
274 
275  cmat rhoA = ptrace(rA, subsysAbar, dims);
276  cmat rhoB = ptrace(rA, subsysBbar, dims);
277  cmat rhoAB = ptrace(rA, subsysABbar, dims);
278 
279  return shannon(rhoA) + shannon(rhoB) - shannon(rhoAB);
280 }

292 double qmutualinfo(const Eigen::MatrixBase <Derived>& A,
293  const std::vector <idx>& subsysA,
294  const std::vector <idx>& subsysB,
295  idx d = 2)
296 {
297  const dyn_mat<typename Derived::Scalar>& rA = A;
298 
299  // error checks
300 
301  // check zero-size
302  if (!internal::_check_nonzero_size(rA))
303  throw Exception("qpp::mutualinfo()", Exception::Type::ZERO_SIZE);
304 
305  idx n =
306  static_cast<idx>(std::llround(std::log2(rA.rows()) /
307  std::log2(d)));
308  std::vector <idx> dims(n, d); // local dimensions vector
309 
310  return qmutualinfo(rA, subsysA, subsysB, dims);
311 }
