491 template<typename Derived>
492 cmat funm(const Eigen::MatrixBase<Derived>& A, cplx (* f)(const cplx&))
493 {
494  const dyn_mat<typename Derived::Scalar>& rA = A;
495 
496  // check zero-size
497  if (!internal::_check_nonzero_size(rA))
498  throw Exception("qpp::funm()", Exception::Type::ZERO_SIZE);
499 
500  // check square matrix
501  if (!internal::_check_square_mat(rA))
502  throw Exception("qpp::funm()", Exception::Type::MATRIX_NOT_SQUARE);
503 
504  Eigen::ComplexEigenSolver<cmat> es(rA.template cast<cplx>());
505  cmat evects = es.eigenvectors();
506  cmat evals = es.eigenvalues();
507  for (idx i = 0; i < static_cast<idx>(evals.rows()); ++i)
508  evals(i) = (*f)(evals(i)); // apply f(x) to each eigenvalue
509 
510  cmat evalsdiag = evals.asDiagonal();
511 
512  return evects * evalsdiag * evects.inverse();
513 }

658 template<typename Derived>
659 cmat spectralpowm(const Eigen::MatrixBase<Derived>& A, const cplx z)
660 {
661  const dyn_mat<typename Derived::Scalar>& rA = A;
662 
663  // check zero-size
664  if (!internal::_check_nonzero_size(rA))
665  throw Exception("qpp::spectralpowm()", Exception::Type::ZERO_SIZE);
666 
667  // check square matrix
668  if (!internal::_check_square_mat(rA))
669  throw Exception("qpp::spectralpowm()",
670  Exception::Type::MATRIX_NOT_SQUARE);
671 
672  // Define A^0 = Id, for z IDENTICALLY zero
673  if (real(z) == 0 && imag(z) == 0)
674  return cmat::Identity(rA.rows(), rA.rows());
675 
676  Eigen::ComplexEigenSolver<cmat> es(rA.template cast<cplx>());
677  cmat evects = es.eigenvectors();
678  cmat evals = es.eigenvalues();
679  for (idx i = 0; i < static_cast<idx>(evals.rows()); ++i)
680  evals(i) = std::pow(evals(i), z);
681 
682  cmat evalsdiag = evals.asDiagonal();
683 
684  return evects * evalsdiag * evects.inverse();
685 }

937 dyn_mat<typename Derived::Scalar> dirsum(const std::vector<Derived>& As)
938 {
939  if (As.size() == 0)
940  throw Exception("qpp::dirsum()", Exception::Type::ZERO_SIZE);
941 
942  for (auto&& it : As)
943  if (!internal::_check_nonzero_size(it))
944  throw Exception("qpp::dirsum()", Exception::Type::ZERO_SIZE);
945 
946  idx total_rows = 0, total_cols = 0;
947  for (idx i = 0; i < As.size(); ++i)
948  {
949  total_rows += static_cast<idx>(As[i].rows());
950  total_cols += static_cast<idx>(As[i].cols());

953  dyn_mat<typename Derived::Scalar>::Zero(total_rows, total_cols);

956  for (idx i = 0; i < As.size(); ++i)
957  {
958  result.block(cur_row, cur_col, As[i].rows(), As[i].cols()) = As[i];
959  cur_row += static_cast<idx>(As[i].rows());
960  cur_col += static_cast<idx>(As[i].cols());
961  }
962 
963  return result;

1169 template<typename Derived>
1170 dyn_mat<typename Derived::Scalar> grams(const std::vector<Derived>& Vs)
1171 {
1172  // check empty list
1173  if (!internal::_check_nonzero_size(Vs))
1174  throw Exception("qpp::grams()", Exception::Type::ZERO_SIZE);
1175 
1176  for (auto&& it : Vs)
1177  if (!internal::_check_nonzero_size(it))
1178  throw Exception("qpp::grams()", Exception::Type::ZERO_SIZE);
1179 
1180  // check that Vs[0] is a column vector
1181  if (!internal::_check_cvector(Vs[0]))
1182  throw Exception("qpp::grams()", Exception::Type::MATRIX_NOT_CVECTOR);
1183 
1184  // now check that all the rest match the size of the first vector
1185  for (auto&& it : Vs)
1186  if (it.rows() != Vs[0].rows() || it.cols() != 1)
1187  throw Exception("qpp::grams()", Exception::Type::DIMS_NOT_EQUAL);
1188 
1189  dyn_mat<typename Derived::Scalar> cut =
1190  dyn_mat<typename Derived::Scalar>::Identity(Vs[0].rows(),
1191  Vs[0].rows());
1192 
1193  dyn_mat<typename Derived::Scalar> vi =
1194  dyn_mat<typename Derived::Scalar>::Zero(Vs[0].rows(), 1);
1195 
1196  std::vector<dyn_mat<typename Derived::Scalar>> outvecs;
1197  // find the first non-zero vector in the list
1198  idx pos = 0;
1199  for (pos = 0; pos < Vs.size(); ++pos)
1200  {
1201  if (norm(Vs[pos]) > eps) // add it as the first element
1202  {
1203  outvecs.push_back(Vs[pos]);
1204  break;
1205  }
1206  }
1207 
1208  // start the process
1209  for (idx i = pos + 1; i < Vs.size(); ++i)
1210  {
1211  cut -= prj(outvecs[i - 1 - pos]);
1212  vi = cut * Vs[i];
1213  outvecs.push_back(vi);
1214  }
1215 
1216  dyn_mat<typename Derived::Scalar> result(Vs[0].rows(), outvecs.size());
1217 
1218  idx cnt = 0;
1219  for (auto&& it : outvecs)
1220  {
1221  double normV = norm(it);
1222  if (normV > eps) // we add only the non-zero vectors
1223  {
1224  result.col(cnt) = it / normV;
1225  cnt++;
1226  }
1227  }
1228 
1229  return result.block(0, 0, Vs[0].rows(), cnt);
1230 }
