406 template<typename Derived>
407 cmat funm(const Eigen::MatrixBase<Derived> &A, cplx (*f)(const cplx &))
408 {
409  const DynMat<typename Derived::Scalar> &rA = A;
410 
411  // check zero-size
412  if (!internal::_check_nonzero_size(rA))
413  throw Exception("qpp::funm()", Exception::Type::ZERO_SIZE);
414 
415  // check square matrix
416  if (!internal::_check_square_mat(rA))
417  throw Exception("qpp::funm()", Exception::Type::MATRIX_NOT_SQUARE);
418 
419  Eigen::ComplexEigenSolver<cmat> es(rA.template cast<cplx>());
420  cmat evects = es.eigenvectors();
421  cmat evals = es.eigenvalues();
422  for (std::size_t i = 0; i < static_cast<std::size_t>(evals.rows()); ++i)
423  evals(i) = (*f)(evals(i)); // apply f(x) to each eigenvalue
424 
425  cmat evalsdiag = evals.asDiagonal();
426 
427  return evects * evalsdiag * evects.inverse();
428 }

572 template<typename Derived>
573 cmat spectralpowm(const Eigen::MatrixBase<Derived> &A, const cplx z)
574 {
575  const DynMat<typename Derived::Scalar> &rA = A;
576 
577  // check zero-size
578  if (!internal::_check_nonzero_size(rA))
579  throw Exception("qpp::spectralpowm()", Exception::Type::ZERO_SIZE);
580 
581  // check square matrix
582  if (!internal::_check_square_mat(rA))
583  throw Exception("qpp::spectralpowm()",
584  Exception::Type::MATRIX_NOT_SQUARE);
585 
586  // Define A^0 = Id, for z IDENTICALLY zero
587  if (real(z) == 0 && imag(z) == 0)
588  return cmat::Identity(rA.rows(), rA.rows());
589 
590  Eigen::ComplexEigenSolver<cmat> es(rA.template cast<cplx>());
591  cmat evects = es.eigenvectors();
592  cmat evals = es.eigenvalues();
593  for (std::size_t i = 0; i < static_cast<std::size_t>(evals.rows()); ++i)
594  evals(i) = std::pow(static_cast<cplx>(evals(i)), static_cast<cplx>(z));
595 
596  cmat evalsdiag = evals.asDiagonal();
597 
598  return evects * evalsdiag * evects.inverse();
599 
600 }

953 template<typename Derived>
954 DynMat<typename Derived::Scalar> grams(const std::vector<Derived> &Vs)
955 {
956  // check empty list
957  if (!internal::_check_nonzero_size(Vs))
958  throw Exception("qpp::grams()", Exception::Type::ZERO_SIZE);
959 
960  for (auto &&it : Vs)
961  if (!internal::_check_nonzero_size(it))
962  throw Exception("qpp::grams()", Exception::Type::ZERO_SIZE);
963 
964  // check that Vs[0] is a column vector
965  if (!internal::_check_col_vector(Vs[0]))
966  throw Exception("qpp::grams()", Exception::Type::MATRIX_NOT_CVECTOR);
967 
968  // now check that all the rest match the size of the first vector
969  for (auto &&it : Vs)
970  if (it.rows() != Vs[0].rows() || it.cols() != 1)
971  throw Exception("qpp::grams()", Exception::Type::DIMS_NOT_EQUAL);
972 
973  DynMat<typename Derived::Scalar> cut =
974  DynMat<typename Derived::Scalar>::Identity(Vs[0].rows(),
975  Vs[0].rows());
976 
977  DynMat<typename Derived::Scalar> vi =
978  DynMat<typename Derived::Scalar>::Zero(Vs[0].rows(), 1);
979 
980  std::vector<DynMat<typename Derived::Scalar>> outvecs;
981  // find the first non-zero vector in the list
982  std::size_t pos = 0;
983  for (pos = 0; pos < Vs.size(); ++pos)
984  {
985  if (norm(Vs[pos]) > eps) // add it as the first element
986  {
987  outvecs.push_back(Vs[pos]);
988  break;
989  }
990  }
991 
992  // start the process
993  for (std::size_t i = pos + 1; i < Vs.size(); ++i)
994  {
995  cut -= prj(outvecs[i - 1 - pos]);
996  vi = cut * Vs[i];
997  outvecs.push_back(vi);
998  }
999 
1000  DynMat<typename Derived::Scalar> result(Vs[0].rows(), outvecs.size());
1001 
1002  std::size_t cnt = 0;
1003  for (auto &&it : outvecs)
1004  {
1005  double normV = norm(it);
1006  if (normV > eps) // we add only the non-zero vectors
1007  {
1008  result.col(cnt) = it / normV;
1009  cnt++;
1010  }
1011  }
1012 
1013  return result.block(0, 0, Vs[0].rows(), cnt);
1014 }
