477 template<typename Derived>
478 cmat funm(const Eigen::MatrixBase <Derived>& A, cplx (* f)(const cplx&))
479 {
480  const dyn_mat<typename Derived::Scalar>& rA = A;
481 
482  // check zero-size
483  if (!internal::_check_nonzero_size(rA))
484  throw Exception("qpp::funm()", Exception::Type::ZERO_SIZE);
485 
486  // check square matrix
487  if (!internal::_check_square_mat(rA))
488  throw Exception("qpp::funm()", Exception::Type::MATRIX_NOT_SQUARE);
489 
490  Eigen::ComplexEigenSolver <cmat> es(rA.template cast<cplx>());
491  cmat evects = es.eigenvectors();
492  cmat evals = es.eigenvalues();
493  for (idx i = 0; i < static_cast<idx>(evals.rows()); ++i)
494  evals(i) = (*f)(evals(i)); // apply f(x) to each eigenvalue
495 
496  cmat evalsdiag = evals.asDiagonal();
497 
498  return evects * evalsdiag * evects.inverse();
499 }

643 template<typename Derived>
644 cmat spectralpowm(const Eigen::MatrixBase <Derived>& A, const cplx z)
645 {
646  const dyn_mat<typename Derived::Scalar>& rA = A;
647 
648  // check zero-size
649  if (!internal::_check_nonzero_size(rA))
650  throw Exception("qpp::spectralpowm()", Exception::Type::ZERO_SIZE);
651 
652  // check square matrix
653  if (!internal::_check_square_mat(rA))
654  throw Exception("qpp::spectralpowm()",
655  Exception::Type::MATRIX_NOT_SQUARE);
656 
657  // Define A^0 = Id, for z IDENTICALLY zero
658  if (real(z) == 0 && imag(z) == 0)
659  return cmat::Identity(rA.rows(), rA.rows());
660 
661  Eigen::ComplexEigenSolver <cmat> es(rA.template cast<cplx>());
662  cmat evects = es.eigenvectors();
663  cmat evals = es.eigenvalues();
664  for (idx i = 0; i < static_cast<idx>(evals.rows()); ++i)
665  evals(i) = std::pow(evals(i), z);
666 
667  cmat evalsdiag = evals.asDiagonal();
668 
669  return evects * evalsdiag * evects.inverse();
670 }

914 template<typename Derived1, typename Derived2>
915 dyn_mat<typename Derived1::Scalar> comm(const Eigen::MatrixBase <Derived1>& A,
916  const Eigen::MatrixBase <Derived2>& B)
917 {
918  const dyn_mat<typename Derived1::Scalar>& rA = A;
919  const dyn_mat<typename Derived2::Scalar>& rB = B;
920 
921  // EXCEPTION CHECKS
922 
923  // check types
924  if (!std::is_same<typename Derived1::Scalar,
925  typename Derived2::Scalar>::value)
926  throw Exception("qpp::comm()", Exception::Type::TYPE_MISMATCH);
927 
928  // check zero-size
929  if (!internal::_check_nonzero_size(rA)
930  || !internal::_check_nonzero_size(rB))
931  throw Exception("qpp::comm()", Exception::Type::ZERO_SIZE);
932 
933  // check square matrices
934  if (!internal::_check_square_mat(rA) || !internal::_check_square_mat(rB))
935  throw Exception("qpp::comm()", Exception::Type::MATRIX_NOT_SQUARE);
936 
937  // check equal dimensions
938  if (rA.rows() != rB.rows())
939  throw Exception("qpp::comm()", Exception::Type::DIMS_NOT_EQUAL);
940 
941  return rA * rB - rB * rA;
942 }

1025 template<typename Derived>
1026 dyn_mat<typename Derived::Scalar> grams(const std::vector <Derived>& Vs)
1027 {
1028  // check empty list
1029  if (!internal::_check_nonzero_size(Vs))
1030  throw Exception("qpp::grams()", Exception::Type::ZERO_SIZE);
1031 
1032  for (auto&& it : Vs)
1033  if (!internal::_check_nonzero_size(it))
1034  throw Exception("qpp::grams()", Exception::Type::ZERO_SIZE);
1035 
1036  // check that Vs[0] is a column vector
1037  if (!internal::_check_col_vector(Vs[0]))
1038  throw Exception("qpp::grams()", Exception::Type::MATRIX_NOT_CVECTOR);
1039 
1040  // now check that all the rest match the size of the first vector
1041  for (auto&& it : Vs)
1042  if (it.rows() != Vs[0].rows() || it.cols() != 1)
1043  throw Exception("qpp::grams()", Exception::Type::DIMS_NOT_EQUAL);
1044 
1045  dyn_mat<typename Derived::Scalar> cut =
1046  dyn_mat<typename Derived::Scalar>::Identity(Vs[0].rows(),
1047  Vs[0].rows());
1048 
1049  dyn_mat<typename Derived::Scalar> vi =
1050  dyn_mat<typename Derived::Scalar>::Zero(Vs[0].rows(), 1);
1051 
1052  std::vector <dyn_mat<typename Derived::Scalar>> outvecs;
1053  // find the first non-zero vector in the list
1054  idx pos = 0;
1055  for (pos = 0; pos < Vs.size(); ++pos)
1056  {
1057  if (norm(Vs[pos]) > eps) // add it as the first element
1058  {
1059  outvecs.push_back(Vs[pos]);
1060  break;
1061  }
1062  }
1063 
1064  // start the process
1065  for (idx i = pos + 1; i < Vs.size(); ++i)
1066  {
1067  cut -= prj(outvecs[i - 1 - pos]);
1068  vi = cut * Vs[i];
1069  outvecs.push_back(vi);
1070  }
1071 
1072  dyn_mat<typename Derived::Scalar> result(Vs[0].rows(), outvecs.size());
1073 
1074  idx cnt = 0;
1075  for (auto&& it : outvecs)
1076  {
1077  double normV = norm(it);
1078  if (normV > eps) // we add only the non-zero vectors
1079  {
1080  result.col(cnt) = it / normV;
1081  cnt++;
1082  }
1083  }
1084 
1085  return result.block(0, 0, Vs[0].rows(), cnt);
1086 }
