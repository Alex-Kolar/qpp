39 std::vector<int> x2contfrac(double x, std::size_t n, std::size_t cut = 1e5)
40 {
41  if (n == 0)
42  throw Exception("qpp::x2contfrac()", Exception::Type::OUT_OF_RANGE);
43 
44  std::vector<int> result;
45 
46  for (std::size_t i = 0; i < n; ++i)
47  {
48  result.push_back(std::floor(x));
49  x = 1. / (x - std::floor(x));

119 std::size_t gcd(std::size_t m, std::size_t n)
120 {
121  if (m == 0 || n == 0)
122  return (std::max(m, n));
123 
124  std::size_t result = 1;
125  while (n)
126  {
127  result = n;
128  n = m % result;
129  m = result;
130  }
131 
132  return result;
133 }

177 {
178  if (ns.size() == 0)
179  throw Exception("qpp::lcm()", Exception::Type::ZERO_SIZE);
180 
181  if (ns.size() == 1) // convention: lcm({n}) = n
182  return ns[0];
183 
184  if (std::find(std::begin(ns), std::end(ns), 0) != std::end(ns))
185  throw Exception("qpp::lcm()", Exception::Type::OUT_OF_RANGE);
