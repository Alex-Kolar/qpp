47 template<typename Derived1, typename Derived2>
48 dyn_mat<typename Derived1::Scalar> applyCTRL(
49  const Eigen::MatrixBase <Derived1>& state,
50  const Eigen::MatrixBase <Derived2>& A,
51  const std::vector <idx>& ctrl,
52  const std::vector <idx>& subsys,
53  const std::vector <idx>& dims)
54 {
55  const dyn_mat<typename Derived1::Scalar>& rstate = state;
56  const dyn_mat<typename Derived2::Scalar>& rA = A;
57 
58  // EXCEPTION CHECKS
59  // check types
60  if (!std::is_same<typename Derived1::Scalar,
61  typename Derived2::Scalar>::value)
62  throw Exception("qpp::applyCTRL()", Exception::Type::TYPE_MISMATCH);
63 
64  // check zero sizes
65  if (!internal::_check_nonzero_size(rA))
66  throw Exception("qpp::applyCTRL()", Exception::Type::ZERO_SIZE);
67 
68  // check zero sizes
69  if (!internal::_check_nonzero_size(rstate))
70  throw Exception("qpp::applyCTRL()", Exception::Type::ZERO_SIZE);
71 
72  // check square matrix for the gate
73  if (!internal::_check_square_mat(rA))
74  throw Exception("qpp::applyCTRL()",
75  Exception::Type::MATRIX_NOT_SQUARE);
76 
77  // check that all control subsystems have the same dimension
78  idx d = ctrl.size() > 0 ? dims[ctrl[0]] : 1;
79  for (idx i = 1; i < ctrl.size(); ++i)
80  if (dims[ctrl[i]] != d)
81  throw Exception("qpp::applyCTRL()",
82  Exception::Type::DIMS_NOT_EQUAL);
83 
84  // check that dimension is valid
85  if (!internal::_check_dims(dims))
86  throw Exception("qpp::applyCTRL()", Exception::Type::DIMS_INVALID);
87 
88  // check subsys is valid w.r.t. dims
89  if (!internal::_check_subsys_match_dims(subsys, dims))
90  throw Exception("qpp::applyCTRL()",
91  Exception::Type::SUBSYS_MISMATCH_DIMS);
92 
93  // check that gate matches the dimensions of the subsys
94  std::vector <idx> subsys_dims(subsys.size());
95  for (idx i = 0; i < subsys.size(); ++i)
96  subsys_dims[i] = dims[subsys[i]];
97  if (!internal::_check_dims_match_mat(subsys_dims, rA))
98  throw Exception("qpp::applyCTRL()",
99  Exception::Type::MATRIX_MISMATCH_SUBSYS);
100 
101  std::vector <idx> ctrlgate = ctrl; // ctrl + gate subsystem vector
102  ctrlgate.insert(std::end(ctrlgate), std::begin(subsys), std::end(subsys));
103  std::sort(std::begin(ctrlgate), std::end(ctrlgate));
104 
105  // check that ctrl + gate subsystem is valid
106  // with respect to local dimensions
107  if (!internal::_check_subsys_match_dims(ctrlgate, dims))
108  throw Exception("qpp::applyCTRL()",
109  Exception::Type::SUBSYS_MISMATCH_DIMS);
110 
111  // END EXCEPTION CHECKS
112 
113  // construct the table of A^i and (A^dagger)^i
114  std::vector <dyn_mat<typename Derived1::Scalar>> Ai;
115  std::vector <dyn_mat<typename Derived1::Scalar>> Aidagger;
116  for (idx i = 0; i < std::max(d, static_cast<idx>(2)); ++i)
117  {
118  Ai.push_back(powm(rA, i));
119  Aidagger.push_back(powm(adjoint(rA), i));
120  }
121 
122  idx D = rstate.rows(); // total dimension
123  idx n = dims.size(); // total number of subsystems
124  idx ctrlsize = ctrl.size(); // dimension of ctrl subsystem
125  idx DA = rA.rows(); // dimension of gate subsystem
126 
127  idx Cdims[maxn]; // local dimensions
128  idx CdimsA[maxn]; // local dimensions
129  idx CdimsCTRLAbar[maxn]; // local dimensions
130 
131  std::vector <idx> ctrlgatebar(n - ctrlgate.size()); // rest
132  std::vector <idx> allsubsys(n); // all subsystems
133  std::iota(std::begin(allsubsys), std::end(allsubsys), 0);
134  // compute the complementary subsystem of ctrlgate w.r.t. dims
135  std::set_difference(std::begin(allsubsys), std::end(allsubsys),
136  std::begin(ctrlgate), std::end(ctrlgate), std::begin(ctrlgatebar));
137 
138  idx DCTRLAbar = 1; // dimension of the rest
139  for (idx i = 0; i < ctrlgatebar.size(); ++i)
140  DCTRLAbar *= dims[ctrlgatebar[i]];
141 
142  for (idx k = 0; k < n; ++k)
143  Cdims[k] = dims[k];
144  for (idx k = 0; k < subsys.size(); ++k)
145  CdimsA[k] = dims[subsys[k]];
146  for (idx k = 0; k < ctrlgatebar.size(); ++k)
147  CdimsCTRLAbar[k] = dims[ctrlgatebar[k]];
148 
149 
150  // worker, computes the coefficient and the index for the ket case
151  // used in #pragma omp parallel for collapse
152  auto coeff_idx_ket = [=](idx _i, idx _m, idx _r)
153  -> std::pair<typename Derived1::Scalar, idx>
154  {
155  idx indx = 0;
156  typename Derived1::Scalar coeff = 0;
157 
158  idx Cmidx[maxn]; // the total multi-index
159  idx CmidxA[maxn];// the gate part multi-index
160  idx CmidxCTRLAbar[maxn];// the rest multi-index
161 
162  // compute the index
163 
164  // set the CTRL part
165  for (idx k = 0; k < ctrl.size(); ++k)
166  {
167  Cmidx[ctrl[k]] = _i;
168  }
169 
170  // set the rest
171  internal::_n2multiidx(_r, n - ctrlgate.size(),
172  CdimsCTRLAbar, CmidxCTRLAbar);
173  for (idx k = 0; k < n - ctrlgate.size(); ++k)
174  {
175  Cmidx[ctrlgatebar[k]] = CmidxCTRLAbar[k];
176  }
177 
178  // set the A part
179  internal::_n2multiidx(_m, subsys.size(), CdimsA, CmidxA);
180  for (idx k = 0; k < subsys.size(); ++k)
181  {
182  Cmidx[subsys[k]] = CmidxA[k];
183  }
184 
185  // we now got the total index
186  indx = internal::_multiidx2n(Cmidx, n, Cdims);
187 
188  // compute the coefficient
189  for (idx _n = 0; _n < DA; ++_n)
190  {
191  internal::_n2multiidx(_n, subsys.size(), CdimsA, CmidxA);
192  for (idx k = 0; k < subsys.size(); ++k)
193  {
194  Cmidx[subsys[k]] = CmidxA[k];
195  }
196  coeff += Ai[_i](_m, _n) *
197  rstate(internal::_multiidx2n(Cmidx, n, Cdims));
198  }
199 
200  return std::make_pair(coeff, indx);
201  };
202 
203  // worker, computes the coefficient and the index
204  // for the density matrix case
205  // used in #pragma omp parallel for collapse
206  auto coeff_idx_rho = [=](idx _i1, idx _m1,
207  idx _r1, idx _i2, idx _m2, idx _r2)
208  -> std::tuple<typename Derived1::Scalar, idx, idx>
209  {
210  idx idxrow = 0;
211  idx idxcol = 0;
212  typename Derived1::Scalar coeff = 0;
213 
214  idx Cmidxrow[maxn]; // the total row multi-index
215  idx Cmidxcol[maxn];// the total col multi-index
216  idx CmidxArow[maxn];// the gate part row multi-index
217  idx CmidxAcol[maxn];// the gate part col multi-index
218  idx CmidxCTRLAbarrow[maxn];// the rest row multi-index
219  idx CmidxCTRLAbarcol[maxn];// the rest col multi-index
220 
221  // compute the ket/bra indexes
222 
223  // set the CTRL part
224  for (idx k = 0; k < ctrl.size(); ++k)
225  {
226  Cmidxrow[ctrl[k]] = _i1;
227  Cmidxcol[ctrl[k]] = _i2;
228  }
229 
230  // set the rest
231  internal::_n2multiidx(_r1, n - ctrlgate.size(),
232  CdimsCTRLAbar, CmidxCTRLAbarrow);
233  internal::_n2multiidx(_r2, n - ctrlgate.size(),
234  CdimsCTRLAbar, CmidxCTRLAbarcol);
235  for (idx k = 0; k < n - ctrlgate.size(); ++k)
236  {
237  Cmidxrow[ctrlgatebar[k]] = CmidxCTRLAbarrow[k];
238  Cmidxcol[ctrlgatebar[k]] = CmidxCTRLAbarcol[k];
239  }
240 
241  // set the A part
242  internal::_n2multiidx(_m1, subsys.size(), CdimsA, CmidxArow);
243  internal::_n2multiidx(_m2, subsys.size(), CdimsA, CmidxAcol);
244  for (idx k = 0; k < subsys.size(); ++k)
245  {
246  Cmidxrow[subsys[k]] = CmidxArow[k];
247  Cmidxcol[subsys[k]] = CmidxAcol[k];
248  }
249 
250  // we now got the total row/col indexes
251  idxrow = internal::_multiidx2n(Cmidxrow, n, Cdims);
252  idxcol = internal::_multiidx2n(Cmidxcol, n, Cdims);
253 
254  // compute the coefficient
255  for (idx _n1 = 0; _n1 < DA; ++_n1)
256  {
257  internal::_n2multiidx(_n1, subsys.size(), CdimsA, CmidxArow);
258  for (idx k = 0; k < subsys.size(); ++k)
259  {
260  Cmidxrow[subsys[k]] = CmidxArow[k];
261  }
262  for (idx _n2 = 0; _n2 < DA; ++_n2)
263  {
264  internal::_n2multiidx(_n2, subsys.size(), CdimsA, CmidxAcol);
265  for (idx k = 0; k < subsys.size(); ++k)
266  {
267  Cmidxcol[subsys[k]] = CmidxAcol[k];
268  }
269  coeff += Ai[_i1](_m1, _n1) *
270  rstate(internal::_multiidx2n(Cmidxrow, n, Cdims),
271  internal::_multiidx2n(Cmidxcol, n, Cdims)) *
272  Aidagger[_i2](_n2, _m2);
273  }
274  }
275 
276  return std::make_tuple(coeff, idxrow, idxcol);
277  };
278 
279  //************ ket ************//
280  if (internal::_check_col_vector(rstate)) // we have a ket
281  {
282  // check that dims match state vector
283  if (!internal::_check_dims_match_cvect(dims, rstate))
284  throw Exception("qpp::applyCTRL()",
285  Exception::Type::DIMS_MISMATCH_CVECTOR);
286  if (D == 1)
287  return rstate;
288 
289  dyn_mat<typename Derived1::Scalar> result = rstate;
290 
291 #pragma omp parallel for collapse(2)
292  for (idx m = 0; m < DA; ++m)
293  for (idx r = 0; r < DCTRLAbar; ++r)
294  {
295  if (ctrlsize == 0) // no control
296  {
297  result(coeff_idx_ket(1, m, r).second) =
298  coeff_idx_ket(1, m, r).first;
299  }
300  else
301  for (idx i = 0; i < d; ++i)
302  {
303  result(coeff_idx_ket(i, m, r).second) =
304  coeff_idx_ket(i, m, r).first;
305  }
306  }
307 
308  return result;
309  }
310  //************ density matrix ************//
311  else if (internal::_check_square_mat(rstate)) // we have a density matrix
312  {
313  // check that dims match state matrix
314  if (!internal::_check_dims_match_mat(dims, rstate))
315  throw Exception("qpp::applyCTRL()",
316  Exception::Type::DIMS_MISMATCH_MATRIX);
317 
318  if (D == 1)
319  return rstate;
320 
321  dyn_mat<typename Derived1::Scalar> result = rstate;
322 
323 #pragma omp parallel for collapse(4)
324  for (idx m1 = 0; m1 < DA; ++m1)
325  for (idx r1 = 0; r1 < DCTRLAbar; ++r1)
326  for (idx m2 = 0; m2 < DA; ++m2)
327  for (idx r2 = 0; r2 < DCTRLAbar; ++r2)
328  if (ctrlsize == 0) // no control
329  {
330  auto coeff_idxes = coeff_idx_rho(1, m1, r1,
331  1, m2, r2);
332  result(std::get<1>(coeff_idxes),
333  std::get<2>(coeff_idxes)) =
334  std::get<0>(coeff_idxes);
335  }
336  else
337  {
338  for (idx i1 = 0; i1 < d; ++i1)
339  for (idx i2 = 0; i2 < d; ++i2)
340  {
341  auto coeff_idxes = coeff_idx_rho(
342  i1, m1, r1,
343  i2, m2, r2);
344  result(std::get<1>(coeff_idxes),
345  std::get<2>(coeff_idxes)) =
346  std::get<0>(coeff_idxes);
347  }
348  }
349 
350  return result;
351  }
352 
353  //************ Exception: not ket nor density matrix ************//
354  else
355  throw Exception("qpp::applyCTRL()",
356  Exception::Type::MATRIX_NOT_SQUARE_OR_CVECTOR);
357 }

521 template<typename Derived>
522 cmat apply(const Eigen::MatrixBase <Derived>& rho,
523  const std::vector <cmat>& Ks)
524 {
525  const cmat& rrho = rho;
526 
527  // EXCEPTION CHECKS
528  if (!internal::_check_nonzero_size(rrho))
529  throw Exception("qpp::apply()", Exception::Type::ZERO_SIZE);
530  if (!internal::_check_square_mat(rrho))
531  throw Exception("qpp::apply()", Exception::Type::MATRIX_NOT_SQUARE);
532  if (Ks.size() == 0)
533  throw Exception("qpp::apply()", Exception::Type::ZERO_SIZE);
534  if (!internal::_check_square_mat(Ks[0]))
535  throw Exception("qpp::apply()", Exception::Type::MATRIX_NOT_SQUARE);
536  if (Ks[0].rows() != rrho.rows())
537  throw Exception("qpp::apply()",
538  Exception::Type::DIMS_MISMATCH_MATRIX);
539  for (auto&& it : Ks)
540  if (it.rows() != Ks[0].rows() || it.cols() != Ks[0].rows())
541  throw Exception("qpp::apply()", Exception::Type::DIMS_NOT_EQUAL);
542 
543  cmat result = cmat::Zero(rrho.rows(), rrho.rows());
544 
545 #pragma omp parallel for
546  for (idx i = 0; i < Ks.size(); ++i)
547  {
548 #pragma omp critical
549  {
550  result += Ks[i] * rrho * adjoint(Ks[i]);
551  }
552  }
553 
554  return result;
555 }

663 cmat super(const std::vector <cmat>& Ks)
664 {
665  // EXCEPTION CHECKS
666  if (Ks.size() == 0)
667  throw Exception("qpp::super()", Exception::Type::ZERO_SIZE);
668  if (!internal::_check_nonzero_size(Ks[0]))
669  throw Exception("qpp::super()", Exception::Type::ZERO_SIZE);
670  if (!internal::_check_square_mat(Ks[0]))
671  throw Exception("qpp::super()", Exception::Type::MATRIX_NOT_SQUARE);
672  for (auto&& it : Ks)
673  if (it.rows() != Ks[0].rows() || it.cols() != Ks[0].rows())
674  throw Exception("qpp::super()", Exception::Type::DIMS_NOT_EQUAL);
675  idx D = static_cast<idx>(Ks[0].rows());
676 
677  cmat result(D * D, D * D);
678  cmat MN = cmat::Zero(D, D);
679  bra A = bra::Zero(D);
680  ket B = ket::Zero(D);
681  cmat EMN = cmat::Zero(D, D);
682 
683 #pragma omp parallel for collapse(2)
684  for (idx m = 0; m < D; ++m)
685  {
