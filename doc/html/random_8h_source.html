282 inline cmat randU(idx D)
283 // ~3 times slower than Toby Cubitt's MATLAB corresponding routine,
284 // because Eigen 3 QR algorithm is not parallelized
285 {
286  if (D == 0)
287  throw Exception("qpp::randU()", Exception::Type::DIMS_INVALID);
288 
289  cmat X(D, D);
290 
291  X = 1 / std::sqrt(2.) * randn < cmat > (D, D);
292  Eigen::HouseholderQR <cmat> qr(X);
293 
294  cmat Q = qr.householderQ();
295  // phase correction so that the resultant matrix is
296  // uniformly distributed according to the Haar measure
297 
298  Eigen::VectorXcd phases = (rand<dmat>(D, 1)).cast<cplx>();
299  for (idx i = 0; i < static_cast<idx>(phases.rows()); ++i)
300  phases(i) = std::exp(2 * pi * 1_i * phases(i));
301 
302  Q = Q * phases.asDiagonal();
303 
304  return Q;
305 }

332 inline std::vector <cmat> randkraus(idx N, idx D)
333 {
334  if (N == 0)
335  throw Exception("qpp::randkraus()", Exception::Type::OUT_OF_RANGE);
336  if (D == 0)
337  throw Exception("qpp::randkraus()", Exception::Type::DIMS_INVALID);
338 
339  std::vector <cmat> result(N);
340  for (idx i = 0; i < N; ++i)
341  result[i] = cmat::Zero(D, D);
342 
343  cmat Fk(D, D);
344  cmat U = randU(N * D);
345 
346 #pragma omp parallel for collapse(3)
347  for (idx k = 0; k < N; ++k)
348  for (idx a = 0; a < D; ++a)
349  for (idx b = 0; b < D; ++b)
350  result[k](a, b) = U(a * N + k, b * N);
351 
352  return result;
353 }
