466  iterator& operator++() {
467  // EXCEPTION CHECKS
468 
469  // protects against incrementing invalid iterators
470  if (qcd_ == nullptr) {
471  throw exception::InvalidIterator(
472  "qpp::QCircuit::iterator::operator++()");
473  }
474 
475  // protect against incrementing an empty circuit iterator
476  if (qcd_->get_step_count() == 0) {
477  throw exception::InvalidIterator(
478  "qpp::QCircuit::iterator::operator++()");
479  }
480 
481  // protects against incrementing past the end
482  if (elem_.ip_ == qcd_->get_step_count()) {
483  throw exception::InvalidIterator(
484  "qpp::QCircuit::iterator::operator++()");
485  }
486  // END EXCEPTION CHECKS
487 
488  // gate step
489  if (elem_.type_ == StepType::GATE) {
490  std::advance(elem_.gates_ip_, 1);
491  }
492  // measurement step
493  else if (elem_.type_ == StepType::MEASUREMENT) {
494  std::advance(elem_.measurements_ip_, 1);
495  }
496  // otherwise
497  else {
498  }
499 
500  // increment the instruction pointer
501  ++elem_.ip_;
502 
503  // if we hit the end
504  if (elem_.ip_ == qcd_->get_step_count()) {
505  elem_.type_ = StepType::NONE;
506  } else {
507  // set the next step type
508  elem_.type_ = qcd_->step_types_[elem_.ip_];
509  }
510 
511  return *this;
512  }
