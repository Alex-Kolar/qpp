861  // implemented the algorithm D (Sec. 3.1) from
862  // http://www.informatik.uni-bremen.de/agra/doc/konf/13_rc_depth_impr.pdf
872  idx get_gate_depth(const std::string& name = "") const {
873  idx result = 0;
874  bool found = false;
875  std::vector<idx> b(nc_ + nq_, 0);
876 
877  for (auto&& step : *this) {
878  if (step.type_ != StepType::GATE)
879  continue;
880 
881  GateStep gate_step = *step.gates_ip_;
882  if (name != "" && gate_step.name_ != name)
883  continue;
884 
885  found = true; // gate was found in the circuit
886 
887  std::vector<idx> ctrl = gate_step.ctrl_;
888  std::vector<idx> target = gate_step.target_;
889  std::vector<idx> ctrl_target;
890  ctrl_target.reserve(ctrl.size() + target.size());
891  ctrl_target.insert(ctrl_target.end(), ctrl.begin(), ctrl.end());
892  ctrl_target.insert(ctrl_target.end(), target.begin(), target.end());
893 
894  bool overlap = false;
895 
896  switch (gate_step.gate_type_) {
897  case GateType::NONE:
898  case GateType::SINGLE:
899  case GateType::TWO:
900  case GateType::THREE:
901  case GateType::CUSTOM:
902  case GateType::FAN:
903  case GateType::SINGLE_CTRL_SINGLE_TARGET:
904  case GateType::SINGLE_CTRL_MULTIPLE_TARGET:
905  case GateType::MULTIPLE_CTRL_SINGLE_TARGET:
906  case GateType::MULTIPLE_CTRL_MULTIPLE_TARGET:
907  case GateType::CUSTOM_CTRL:
908  // apply (ctrl) gate
909  for (auto&& i : ctrl_target)
910  b[nc_ + i] += 1;
911  // check whether gates overlap
912  for (auto&& elem : b)
913  if (elem == 2)
914  overlap = true;
915  if (overlap) {
916  // reset the b vector
917  b = std::vector<idx>(nc_ + nq_, 0);
918  // set to 1 the locations of the last gate
919  for (auto&& i : ctrl_target)
920  b[nc_ + i] = 1;
921  ++result;
922  }
923  break;
924  case GateType::SINGLE_cCTRL_SINGLE_TARGET:
925  case GateType::SINGLE_cCTRL_MULTIPLE_TARGET:
926  case GateType::MULTIPLE_cCTRL_SINGLE_TARGET:
927  case GateType::MULTIPLE_cCTRL_MULTIPLE_TARGET:
928  case GateType::CUSTOM_cCTRL:
929  // apply classical ctrl
930  for (auto&& i : ctrl)
931  b[i] += 1;
932  // apply gate
933  for (auto&& i : target)
934  b[nc_ + i] += 1;
935  // check whether gates overlap
936  for (auto&& elem : b) {
937  if (elem == 2) {
938  overlap = true;
939  }
940  }
941  if (overlap) {
942  // reset the b vector
943  b = std::vector<idx>(nc_ + nq_, 0);
944  // set to 1 the locations of the last gate
945  for (auto&& i : ctrl)
946  b[i] = 1;
947  for (auto&& i : target)
948  b[nc_ + i] = 1;
949  ++result;
950  }
951  break;
952  } // end switch
953  } // end for
954 
955  return found ? result + 1 : 0;
956  }

1019  QCircuit& gate(const cmat& U, idx i, std::string name = "") {
1020  // EXCEPTION CHECKS
1021 
1022  try {
1023  // check valid target
1024  if (i >= nq_)
1025  throw exception::OutOfRange("qpp::QCircuit::gate()");
1026  // check not measured before
1027  if (get_measured(i))
1028  throw exception::QuditAlreadyMeasured("qpp::QCircuit::gate()");
1029 
1030  // check square matrix for the gate
1031  if (!internal::check_square_mat(U))
1032  throw exception::MatrixNotSquare("qpp::QCircuit::gate()");
1033  // check correct dimension
1034  if (static_cast<idx>(U.rows()) != d_)
1035  throw exception::DimsMismatchMatrix("qpp::QCircuit::gate()");
1036  } catch (exception::Exception&) {
1037  std::cerr << "At STEP " << get_step_count() << "\n";
1038  throw;
1039  }
1040  // END EXCEPTION CHECKS
1041 
1042  if (name == "")
1043  name = qpp::Gates::get_instance().get_name(U);
1044  std::size_t hashU = hash_eigen(U);
1045  add_hash_(U, hashU);
1046  gates_.emplace_back(GateType::SINGLE, hashU, std::vector<idx>{},
1047  std::vector<idx>{i}, name);
1048  step_types_.emplace_back(StepType::GATE);
1049  ++count_[name];
1050 
1051  return *this;
1052  }

1063  QCircuit& gate(const cmat& U, idx i, idx j, std::string name = "") {
1064  // EXCEPTION CHECKS
1065 
1066  try {
1067  // check valid target
1068  if (i >= nq_ || j >= nq_ || i == j)
1069  throw exception::OutOfRange("qpp::QCircuit::gate()");
1070  if (get_measured(i) || get_measured(j))
1071  throw exception::QuditAlreadyMeasured("qpp::QCircuit::gate()");
1072 
1073  // check square matrix for the gate
1074  if (!internal::check_square_mat(U))
