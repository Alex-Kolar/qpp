495  throw qpp::exception::InvalidIterator(

501  throw qpp::exception::InvalidIterator(

508  std::advance(elem_.gates_ip_, 1);

512  std::advance(elem_.measurements_ip_, 1);

514  // otherwise

516  }
517 
518  // increment the instruction pointer
519  ++elem_.ip_;
520 
521  // if we hit the end

523  elem_.type_ = StepType::NONE;
524  } else {
525  // set the next step type
526  elem_.type_ = qcd_->step_types_[elem_.ip_];
