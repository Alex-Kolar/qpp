265  iterator& operator++() {
266  // EXCEPTION CHECKS
267 
268  // protects against incrementing invalid iterators
269  if (qcd_ == nullptr) {
270  throw qpp::exception::InvalidIterator(
271  "qpp::QCircuitDescription::iterator::operator++()");
272  }
273 
274  // protects against incrementing past the end
275  if (elem_.ip_ == qcd_->get_steps_count()) {
276  throw qpp::exception::InvalidIterator(
277  "qpp::QCircuitDescription::iterator::operator++()");
278  }
279 
280  // protect against incrementing an empty circuit iterator
281  if (elem_.q_ip_ == idx_infty && elem_.m_ip_ == idx_infty) {
282  throw qpp::exception::InvalidIterator(
283  "qpp::QCircuitDescription::iterator::operator++()");
284  }
285  // END EXCEPTION CHECKS
286 
287  // increment the instruction pointer
288  ++elem_.ip_;
289 
290  // only measurements, no gates
291  if (elem_.q_ip_ == idx_infty) {
292  elem_.is_measurement_ = true;
293  ++elem_.m_ip_;
294  return *this;
295  }
296 
297  // only gates, no measurements
298  if (elem_.m_ip_ == idx_infty) {
299  elem_.is_measurement_ = false;
300  ++elem_.q_ip_;
301  return *this;
302  }
303 
304  // current step is a measurement
305  if (elem_.m_ip_ < qcd_->get_measurement_count() &&
306  qcd_->measurement_steps_[elem_.m_ip_] == elem_.q_ip_) {
307  // next step is a measurement
308  if (elem_.m_ip_ + 1 < qcd_->get_measurement_count() &&
309  qcd_->measurement_steps_[elem_.m_ip_ + 1] == elem_.q_ip_) {
310  elem_.is_measurement_ = true;
311  ++elem_.m_ip_;
312  } else
313  // next step is a gate
314  {
315  elem_.is_measurement_ = false;
316  ++elem_.m_ip_;
317  }
318  } else
319  // current step is a gate
320  {
321  // next step is a measurement
322  if (elem_.m_ip_ < qcd_->get_measurement_count() &&
323  qcd_->measurement_steps_[elem_.m_ip_] == elem_.q_ip_ + 1) {
324  elem_.is_measurement_ = true;
325  ++elem_.q_ip_;
326  } else
327  // next step is a gate
328  {
329  elem_.is_measurement_ = false;
330  ++elem_.q_ip_;
331  }
332  }
333 
334  return *this;
335  }
