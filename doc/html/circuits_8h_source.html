442  iterator& operator++() {
443  // EXCEPTION CHECKS
444 
445  // protects against incrementing invalid iterators
446  if (qcd_ == nullptr) {
447  throw exception::InvalidIterator(
448  "qpp::QCircuit::iterator::operator++()");
449  }
450 
451  // protect against incrementing an empty circuit iterator
452  if (qcd_->get_step_count() == 0) {
453  throw exception::InvalidIterator(
454  "qpp::QCircuit::iterator::operator++()");
455  }
456 
457  // protects against incrementing past the end
458  if (elem_.ip_ == qcd_->get_step_count()) {
459  throw exception::InvalidIterator(
460  "qpp::QCircuit::iterator::operator++()");
461  }
462  // END EXCEPTION CHECKS
463 
464  // Gate step
465  if (elem_.type_ == StepType::GATE) {
466  std::advance(elem_.gates_ip_, 1);
467  }
468  // Measurement step
469  else if (elem_.type_ == StepType::MEASUREMENT) {
470  std::advance(elem_.measurements_ip_, 1);
471  }
472  // otherwise
473  else {
474  }
475 
476  // increment the instruction pointer
477  ++elem_.ip_;
478 
479  // if we hit the end
480  if (elem_.ip_ == qcd_->get_step_count()) {
481  elem_.type_ = StepType::NONE;
482  } else {
483  // set the next step type
484  elem_.type_ = qcd_->step_types_[elem_.ip_];
485  }
486 
487  return *this;
488  }
