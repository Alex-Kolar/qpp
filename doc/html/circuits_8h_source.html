510  iterator& operator++() {
511  // EXCEPTION CHECKS
512 
513  // protects against incrementing invalid iterators
514  if (qc_ == nullptr) {
515  throw exception::InvalidIterator(
516  "qpp::QCircuit::iterator::operator++()");
517  }
518 
519  // protects against incrementing an empty circuit iterator
520  if (qc_->get_step_count() == 0) {
521  throw exception::InvalidIterator(
522  "qpp::QCircuit::iterator::operator++()");
523  }
524 
525  // protects against incrementing past the end
526  if (elem_.ip_ == qc_->get_step_count()) {
527  throw exception::InvalidIterator(
528  "qpp::QCircuit::iterator::operator++()");
529  }
530  // END EXCEPTION CHECKS
531 
532  // gate step
533  if (elem_.type_ == StepType::GATE) {
534  std::advance(elem_.gates_ip_, 1);
535  }
536  // measurement step
537  else if (elem_.type_ == StepType::MEASUREMENT) {
538  std::advance(elem_.measurements_ip_, 1);
539  }
540  // otherwise
541  else {
542  }
543 
544  // increment the instruction pointer
545  ++elem_.ip_;
546 
547  // if we hit the end
548  if (elem_.ip_ == qc_->get_step_count()) {
549  elem_.type_ = StepType::NONE;
550  } else {
551  // set the next step type
552  elem_.type_ = qc_->step_types_[elem_.ip_];
553  }
554 
555  return *this;
556  }

932  QCircuit& gate(const cmat& U, idx i, std::string name = "") {
933  // EXCEPTION CHECKS
934 
935  try {
936  // check valid target
937  if (i >= nq_)
938  throw exception::OutOfRange("qpp::QCircuit::gate()");
939  // check not measured before
940  if (get_measured(i))
941  throw exception::QuditAlreadyMeasured("qpp::QCircuit::gate()");
942 
943  // check square matrix for the gate
944  if (!internal::check_square_mat(U))
945  throw exception::MatrixNotSquare("qpp::QCircuit::gate()");
946  // check correct dimension
947  if (static_cast<idx>(U.rows()) != d_)
948  throw exception::DimsMismatchMatrix("qpp::QCircuit::gate()");
949  } catch (exception::Exception&) {
950  std::cerr << "At STEP " << get_step_count() << "\n";
951  throw;
952  }
953  // END EXCEPTION CHECKS
954 
955  if (name == "")
956  name = qpp::Gates::get_instance().get_name(U);
957  std::size_t hashU = hash_eigen(U);
958  add_hash_(U, hashU);
959  gates_.emplace_back(GateType::SINGLE, hashU, std::vector<idx>{},
960  std::vector<idx>{i}, name);
961  step_types_.push_back(StepType::GATE);
962  ++count_[name];
