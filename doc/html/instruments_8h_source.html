49 template<typename Derived>
50 std::tuple<idx, std::vector<double>, std::vector<cmat>>
51 measure(
52  const Eigen::MatrixBase<Derived>& A,
53  const std::vector<cmat>& Ks,
54  const std::vector<idx>& subsys,
55  const std::vector<idx>& dims)
56 {
57  const cmat& rA = A;
58 
59  // EXCEPTION CHECKS
60 
61  // check zero-size
62  if (!internal::_check_nonzero_size(rA))
63  throw Exception("qpp::measure()", Exception::Type::ZERO_SIZE);
64 
65  // check that dimension is valid
66  if (!internal::_check_dims(dims))
67  throw Exception("qpp::measure()", Exception::Type::DIMS_INVALID);
68 
69  // check that dims match rho matrix
70  if (!internal::_check_dims_match_mat(dims, rA))
71  throw Exception("qpp::measure()",
72  Exception::Type::DIMS_MISMATCH_MATRIX);
73 
74  // check subsys is valid w.r.t. dims
75  if (!internal::_check_subsys_match_dims(subsys, dims))
76  throw Exception("qpp::measure()",
77  Exception::Type::SUBSYS_MISMATCH_DIMS);
78 
79  std::vector<idx> subsys_dims(subsys.size());
80  for (idx i = 0; i < subsys.size(); ++i)
81  subsys_dims[i] = dims[subsys[i]];
82 
83  idx D = 1;
84  for (auto&& it: dims)
85  D *= it;
86 
87  idx Dsubsys = 1;
88  for (auto&& it: subsys_dims)
89  Dsubsys *= it;
90 
91  idx Dbar = D / Dsubsys;
92 
93  // check the Kraus operators
94  if (Ks.size() == 0)
95  throw Exception("qpp::measure()",
96  Exception::Type::ZERO_SIZE);
97  if (!internal::_check_square_mat(Ks[0]))
98  throw Exception("qpp::measure()",
99  Exception::Type::MATRIX_NOT_SQUARE);
100  if (Dsubsys != static_cast<idx>(Ks[0].rows()))
101  throw Exception("qpp::measure()",
102  Exception::Type::DIMS_MISMATCH_MATRIX);
103  for (auto&& it : Ks)
104  if (it.rows() != Ks[0].rows() || it.cols() != Ks[0].rows())
105  throw Exception("qpp::measure()",
106  Exception::Type::DIMS_NOT_EQUAL);
107  // END EXCEPTION CHECKS
108 
109  // probabilities
110  std::vector<double> prob(Ks.size());
111  // resulting states
112  std::vector<cmat> outstates(Ks.size());
113 
114  if (internal::_check_square_mat(rA)) // square matrix
115  {
116  for (idx i = 0; i < Ks.size(); ++i)
117  {
118  outstates[i] = cmat::Zero(Dbar, Dbar);
119  cmat tmp = apply(rA, Ks[i], subsys, dims);
120  tmp = ptrace(tmp, subsys, dims);
121  prob[i] = std::abs(trace(tmp)); // probability
122  if (prob[i] > eps)
123  // normalized output state
124  // corresponding to measurement result i
125  outstates[i] = tmp / prob[i];
126  }
127  }
128  else if (internal::_check_col_vector(rA)) // column vector
129  {
130  for (idx i = 0; i < Ks.size(); ++i)
131  {
132  outstates[i] = cmat::Zero(Dbar, Dbar);
133  ket tmp = apply(rA, Ks[i], subsys, dims);
134  prob[i] = std::pow(norm(tmp), 2);
135  if (prob[i] > eps)
136  outstates[i] = ptrace(prj(tmp), subsys, dims);
137  }
138  }
139  else
140  throw Exception("qpp::measure()",
141  Exception::Type::MATRIX_NOT_SQUARE_OR_CVECTOR);
142 
143  // sample from the probability distribution
144  std::discrete_distribution<idx> dd(std::begin(prob),
145  std::end(prob));
146  idx result = dd(RandomDevices::get_instance()._rng);
147 
148  return std::make_tuple(result, prob, outstates);
149 }

364 template<typename Derived>
365 std::tuple<idx, std::vector<double>, std::vector<cmat>> measure(
366  const Eigen::MatrixBase<Derived>& A, const std::vector<cmat>& Ks)
367 {
368  const dyn_mat<typename Derived::Scalar>& rA = A;
369 
370  // EXCEPTION CHECKS
371  // check zero-size
372  if (!internal::_check_nonzero_size(rA))
373  throw Exception("qpp::measure()", Exception::Type::ZERO_SIZE);
374 
375  // check the Kraus operators
376  if (Ks.size() == 0)
377  throw Exception("qpp::measure()", Exception::Type::ZERO_SIZE);
378  if (!internal::_check_square_mat(Ks[0]))
379  throw Exception("qpp::measure()", Exception::Type::MATRIX_NOT_SQUARE);
380  if (Ks[0].rows() != rA.rows())
381  throw Exception("qpp::measure()",
382  Exception::Type::DIMS_MISMATCH_MATRIX);
383  for (auto&& it : Ks)
384  if (it.rows() != Ks[0].rows() || it.cols() != Ks[0].rows())
385  throw Exception("qpp::measure()", Exception::Type::DIMS_NOT_EQUAL);
386  // END EXCEPTION CHECKS
387 
388  // probabilities
389  std::vector<double> prob(Ks.size());
390  // resulting states
391  std::vector<cmat> outstates(Ks.size());
392 
393  if (internal::_check_square_mat(rA)) // square matrix
394  {
395  for (idx i = 0; i < Ks.size(); ++i)
396  {
397  outstates[i] = cmat::Zero(rA.rows(), rA.rows());
398  cmat tmp = Ks[i] * rA * adjoint(Ks[i]); // un-normalized;
399  prob[i] = std::abs(trace(tmp)); // probability
400  if (prob[i] > eps)
401  outstates[i] = tmp / prob[i]; // normalized
402  }
403  }
404  else if (internal::_check_col_vector(rA)) // column vector
405  {
406  for (idx i = 0; i < Ks.size(); ++i)
407  {
408  outstates[i] = ket::Zero(rA.rows());
409  ket tmp = Ks[i] * rA; // un-normalized;
410  // probability
411  prob[i] = std::abs((adjoint(tmp) * tmp).value());
412  if (prob[i] > eps)
413  outstates[i] = tmp / std::sqrt(prob[i]); // normalized
414  }
415  }
416  else
417  throw Exception("qpp::measure()",
418  Exception::Type::MATRIX_NOT_SQUARE_OR_CVECTOR);
419 
420  // sample from the probability distribution
421  std::discrete_distribution<idx> dd(std::begin(prob),
422  std::end(prob));
423  idx result = dd(RandomDevices::get_instance()._rng);
424 
425  return std::make_tuple(result, prob, outstates);
426 }
