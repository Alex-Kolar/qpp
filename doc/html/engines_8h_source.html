55  void set_measured_(idx i) {
56  if (get_measured(i))
57  throw exception::QuditAlreadyMeasured(

285  virtual void execute(const QCircuit::iterator::value_type& elem) {
286  // EXCEPTION CHECKS
287 
288  // iterator must point to the same quantum circuit
289  if (elem.value_type_qc_ != qc_)
290  throw exception::InvalidIterator("qpp::QEngine::execute()");
291  // the rest of exceptions are caught by the iterator::operator*()
292  // END EXCEPTION CHECKS
293 
294  auto h_tbl = qc_->get_cmat_hash_tbl_();
295 
296  // gate step
297  if (elem.type_ == QCircuit::StepType::GATE) {
298  auto gates = qc_->get_gates_();
299 
300  idx q_ip =
301  std::distance(std::begin(qc_->get_gates_()), elem.gates_ip_);
302 
303  std::vector<idx> ctrl_rel_pos;
304  std::vector<idx> target_rel_pos =
305  get_relative_pos_(gates[q_ip].target_);
306 
307  switch (gates[q_ip].gate_type_) {
308  case QCircuit::GateType::NONE:
309  break;
310  case QCircuit::GateType::SINGLE:
311  case QCircuit::GateType::TWO:
312  case QCircuit::GateType::THREE:
313  case QCircuit::GateType::CUSTOM:
314  psi_ = apply(psi_, h_tbl[gates[q_ip].gate_hash_],
315  target_rel_pos, qc_->get_d());
316  break;
317  case QCircuit::GateType::FAN:
318  for (idx m = 0; m < gates[q_ip].target_.size(); ++m)
319  psi_ = apply(psi_, h_tbl[gates[q_ip].gate_hash_],
320  {target_rel_pos[m]}, qc_->get_d());
321  break;
322  case QCircuit::GateType::SINGLE_CTRL_SINGLE_TARGET:
323  case QCircuit::GateType::SINGLE_CTRL_MULTIPLE_TARGET:
324  case QCircuit::GateType::MULTIPLE_CTRL_SINGLE_TARGET:
325  case QCircuit::GateType::MULTIPLE_CTRL_MULTIPLE_TARGET:
326  case QCircuit::GateType::CUSTOM_CTRL:
327  ctrl_rel_pos = get_relative_pos_(gates[q_ip].ctrl_);
328  psi_ = applyCTRL(psi_, h_tbl[gates[q_ip].gate_hash_],
329  ctrl_rel_pos, target_rel_pos, qc_->get_d());
330  break;
331  case QCircuit::GateType::SINGLE_cCTRL_SINGLE_TARGET:
332  case QCircuit::GateType::SINGLE_cCTRL_MULTIPLE_TARGET:
333  case QCircuit::GateType::MULTIPLE_cCTRL_SINGLE_TARGET:
334  case QCircuit::GateType::MULTIPLE_cCTRL_MULTIPLE_TARGET:
335  case QCircuit::GateType::CUSTOM_cCTRL:
336  if (dits_.size() == 0) {
337  psi_ = apply(psi_, h_tbl[gates[q_ip].gate_hash_],
338  target_rel_pos, qc_->get_d());
339  } else {
340  bool should_apply = true;
341  idx first_dit = dits_[(gates[q_ip].ctrl_)[0]];
342  for (idx m = 0; m < gates[q_ip].ctrl_.size(); ++m) {
343  if (dits_[(gates[q_ip].ctrl_)[m]] != first_dit) {
344  should_apply = false;
345  break;
346  }
347  }
348  if (should_apply) {
349  psi_ = apply(
350  psi_,
351  powm(h_tbl[gates[q_ip].gate_hash_], first_dit),
352  target_rel_pos, qc_->get_d());
353  }
354  }
355  break;
356  } // end switch on gate type
357  } // end if gate step
358 
359  // measurement step
360  else if (elem.type_ == QCircuit::StepType::MEASUREMENT) {
361  auto measurements = qc_->get_measurements_();
362  idx m_ip = std::distance(std::begin(qc_->get_measurements_()),
363  elem.measurements_ip_);
364 
365  std::vector<idx> target_rel_pos =
366  get_relative_pos_(measurements[m_ip].target_);
367 
368  std::vector<idx> resZ;
369  double probZ;
370 
371  idx mres = 0;
372  std::vector<double> probs;
373  std::vector<cmat> states;
374 
375  switch (measurements[m_ip].measurement_type_) {
376  case QCircuit::MeasureType::NONE:
377  break;
378  case QCircuit::MeasureType::MEASURE_Z:
379  std::tie(resZ, probZ, psi_) =
380  measure_seq(psi_, target_rel_pos, qc_->get_d());
381  dits_[measurements[m_ip].c_reg_] = resZ[0];
382  probs_[measurements[m_ip].c_reg_] = probZ;
383  set_measured_(measurements[m_ip].target_[0]);
384  break;
385  case QCircuit::MeasureType::MEASURE_V:
386  std::tie(mres, probs, states) =
387  measure(psi_, h_tbl[measurements[m_ip].mats_hash_[0]],
388  target_rel_pos, qc_->get_d());
389  psi_ = states[mres];
390  dits_[measurements[m_ip].c_reg_] = mres;
391  probs_[measurements[m_ip].c_reg_] = probs[mres];
392  set_measured_(measurements[m_ip].target_[0]);
393  break;
394  case QCircuit::MeasureType::MEASURE_V_MANY:
395  std::tie(mres, probs, states) =
396  measure(psi_, h_tbl[measurements[m_ip].mats_hash_[0]],
397  target_rel_pos, qc_->get_d());
398  psi_ = states[mres];
399  dits_[measurements[m_ip].c_reg_] = mres;
400  probs_[measurements[m_ip].c_reg_] = probs[mres];
401  for (auto&& elem : measurements[m_ip].target_)
402  set_measured_(elem);
403  break;
404  } // end switch on measurement type
405  } // end else if measurement step
406 
407  // no-op
408  else if (elem.type_ == QCircuit::StepType::NOP) {
409  }
410 
411  // otherwise
412  else {
413  }
414  }

493 class QNoisyEngine : public QEngine {
494  const NoiseModel noise_;
495  std::vector<std::vector<idx>> noise_results_;
496  public:
503  explicit QNoisyEngine(const QCircuit& qc, const NoiseModel& noise)
504  : QEngine{qc}, noise_{noise}, noise_results_(qc.get_step_count()) {
505  // EXCEPTION CHECKS
506 
507  // check noise has the correct dimensionality
508  if (qc.get_d() != noise.get_d())
509  throw exception::DimsNotEqual("qpp::QNoisyEngine::QNoisyEngine()");
510  // END EXCEPTION CHECKS
511  }
517  void execute(const QCircuit::iterator::value_type& elem) override {
518  // get the relative position of the target
519  std::vector<idx> target_rel_pos = get_relative_pos_(get_non_measured());
520  // apply the noise
521  for (auto&& i : target_rel_pos) {
522  psi_ = noise_(psi_, i);
523  // record the Kraus operator that occured
524  noise_results_[elem.ip_].emplace_back(noise_.get_last_idx());
525  }
526  // execute the circuit step
527  QEngine::execute(elem);
528  }
534  void execute(const QCircuit::iterator& it) override { execute(*it); }
535
